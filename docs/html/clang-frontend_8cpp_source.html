41 /// ApplyQAOverride - Apply a list of edits to the input argument lists.
42 ///
43 /// The input string is a space separate list of edits to perform,
44 /// they are applied in order to the input argument lists. Edits
45 /// should be one of the following forms:
46 ///
47 /// '#': Silence information about the changes to the command line arguments.
48 ///
49 /// '^': Add FOO as a new argument at the beginning of the command line.
50 ///
51 /// '+': Add FOO as a new argument at the end of the command line.
52 ///
53 /// 's/XXX/YYY/': Substitute the regular expression XXX with YYY in the command
54 /// line.
55 ///
56 /// 'xOPTION': Removes all instances of the literal argument OPTION.
57 ///
58 /// 'XOPTION': Removes all instances of the literal argument OPTION,
59 /// and the following argument.
60 ///
61 /// 'Ox': Removes all flags matching 'O' or 'O[sz0-9]' and adds 'Ox'
62 /// at the end of the command line.
63 ///
64 /// \param OS - The stream to write edit information to.
65 /// \param Args - The vector of command line arguments.
66 /// \param Edit - The override command to perform.
67 /// \param SavedStrings - Set to use for storing string representations.
68 static void ApplyOneQAOverride(raw_ostream &OS,
69  SmallVectorImpl<const char *> &Args,
70  StringRef Edit,
71  std::set<str> &SavedStrings) {
72  // This does not need to be efficient.
73  if (Edit[0] == '^') {
74  const char *Str = GetStableCStr(SavedStrings, Edit.substr(1));
75  OS << "### Adding argument " << Str << " at beginning\n";
76  Args.insert(Args.begin() + 1, Str);
77  } else if (Edit[0] == '+') {
78  const char *Str = GetStableCStr(SavedStrings, Edit.substr(1));
79  OS << "### Adding argument " << Str << " at end\n";
80  Args.push_back(Str);
81  } else if (Edit[0] == 's' && Edit[1] == '/' && Edit.endswith("/")
82  && Edit.slice(2, Edit.size() - 1).find('/') != StringRef::npos) {
83  StringRef MatchPattern = Edit.substr(2).split('/').first;
84  StringRef ReplPattern = Edit.substr(2).split('/').second;
85  ReplPattern = ReplPattern.slice(0, ReplPattern.size() - 1);
86 
87  for (unsigned i = 1, e = (unsigned) Args.size(); i != e; ++i) {
88  // Ignore end-of-line response file markers
89  if (Args[i] == nullptr) {
90  continue;
91  }
92  str Repl = llvm::Regex(MatchPattern).sub(ReplPattern, Args[i]);
93 
94  if (Repl != Args[i]) {
95  OS << "### Replacing '" << Args[i] << "' with '" << Repl << "'\n";
96  Args[i] = GetStableCStr(SavedStrings, Repl);
97  }
98  }
99  } else if (Edit[0] == 'x' || Edit[0] == 'X') {
100  auto Option = Edit.substr(1);
101  for (unsigned i = 1; i < Args.size();) {
102  if (Option == Args[i]) {
103  OS << "### Deleting argument " << Args[i] << '\n';
104  Args.erase(Args.begin() + i);
105  if (Edit[0] == 'X') {
106  if (i < Args.size()) {
107  OS << "### Deleting argument " << Args[i] << '\n';
108  Args.erase(Args.begin() + i);
109  } else {
110  OS << "### Invalid X edit, end of command line!\n";
111  }
112  }
113  } else {
114  ++i;
115  }
116  }
117  } else if (Edit[0] == 'O') {
118  for (unsigned i = 1; i < Args.size();) {
119  const char *A = Args[i];
120  // Ignore end-of-line response file markers
121  if (A == nullptr) {
122  continue;
123  }
124  if (A[0] == '-' && A[1] == 'O'
125  && (A[2] == '\0' || (A[3] == '\0' && (A[2] == 's' || A[2] == 'z' || ('0' <= A[2] && A[2] <= '9'))))) {
126  OS << "### Deleting argument " << Args[i] << '\n';
127  Args.erase(Args.begin() + i);
128  } else {
129  ++i;
130  }
131  }
132  OS << "### Adding argument " << Edit << " at end\n";
133  Args.push_back(GetStableCStr(SavedStrings, '-' + Edit.str()));
134  } else {
135  OS << "### Unrecognized edit: " << Edit << "\n";
136  }
137 }

139 /// ApplyQAOverride - Apply a comma separate list of edits to the
140 /// input argument lists. See ApplyOneQAOverride.
141 static void ApplyQAOverride(SmallVectorImpl<const char *> &Args, const char *OverrideStr, std::set<str> &SavedStrings) {
142  raw_ostream *OS = &llvm::errs();
143 
144  if (OverrideStr[0] == '#') {
145  ++OverrideStr;
146  OS = &llvm::nulls();
147  }
148 
149  *OS << "### CCC_OVERRIDE_OPTIONS: " << OverrideStr << "\n";
150 
151  // This does not need to be efficient.
152 
153  const char *S = OverrideStr;
154  while (*S) {
155  const char *End = ::strchr(S, ' ');
156  if (!End) {
157  End = S + strlen(S);
158  }
159  if (End != S) {
160  ApplyOneQAOverride(*OS, Args, str(S, End), SavedStrings);
161  }
162  S = End;
163  if (*S != '\0') {
164  ++S;
165  }
166  }
167 }

172 static void insertTargetAndModeArgs(const ParsedClangName &NameParts,
173  SmallVectorImpl<const char *> &ArgVector,
174  std::set<str> &SavedStrings) {
175  // Put target and mode arguments at the start of argument list so that
176  // arguments specified in command line could override them. Avoid putting
177  // them at index 0, as an option like '-cc1' must remain the first.
178  int InsertionPoint = 0;
179  if (ArgVector.size() > 0) {
180  ++InsertionPoint;
181  }
182 
183  if (NameParts.DriverMode) {
184  // Add the mode flag to the arguments.
185  ArgVector.insert(ArgVector.begin() + InsertionPoint, GetStableCStr(SavedStrings, NameParts.DriverMode));
186  }
187 
188  if (NameParts.TargetIsValid) {
189  const char *arr[] = {"-target", GetStableCStr(SavedStrings, NameParts.TargetPrefix)};
190  ArgVector.insert(ArgVector.begin() + InsertionPoint, std::begin(arr), std::end(arr));
191  }

274 static int ExecuteCC1Tool(SmallVectorImpl<const char *> &ArgV) {
275  // If we call the cc1 tool from the clangDriver library (through
276  // Driver::CC1Main), we need to clean up the options usage count. The options
277  // are currently global, and they might have been used previously by the
278  // driver.
279  llvm::cl::ResetAllOptionOccurrences();
280 
281  llvm::BumpPtrAllocator A;
282  llvm::StringSaver Saver(A);
283  llvm::cl::ExpandResponseFiles(Saver, &llvm::cl::TokenizeGNUCommandLine, ArgV,
284  /*MarkEOLs=*/false);
285  StringRef Tool = ArgV[1];
286  void *GetExecutablePathVP = (void *) (intptr_t) GetExecutablePath;
287  if (Tool == "-cc1") {
288  return cc1_main(makeArrayRef(ArgV).slice(2), ArgV[0], GetExecutablePathVP);
289  }
290  if (Tool == "-cc1as") {
291  return cc1as_main(makeArrayRef(ArgV).slice(2), ArgV[0], GetExecutablePathVP);
292  }
293  // Reject unknown tools.
294  llvm::errs() << "error: unknown integrated tool '" << Tool << "'. " << "Valid tools include '-cc1' and '-cc1as'.\n";
295  return 1;
296 }

298 int main0(int argc_, const char **argv_) {
299  noteBottomOfStack();
300  llvm::InitLLVM X(argc_, argv_);
301  size_t argv_offset = (strcmp(argv_[1], "-cc1") == 0 || strcmp(argv_[1], "-cc1as") == 0) ? 0 : 1;
302  SmallVector<const char *, 256> argv(argv_ + argv_offset, argv_ + argc_);
303 
304  if (llvm::sys::Process::FixupStandardFileDescriptors()) {
305  return 1;
306  }
307 
308  llvm::InitializeAllTargets();
309  auto TargetAndMode = ToolChain::getTargetAndModeFromProgramName(argv[0]);
310 
311  llvm::BumpPtrAllocator A;
312  llvm::StringSaver Saver(A);
313 
314  // Parse response files using the GNU syntax, unless we're in CL mode. There
315  // are two ways to put clang in CL compatibility mode: argv[0] is either
316  // clang-cl or cl, or --driver-mode=cl is on the command line. The normal
317  // command line parsing can't happen until after response file parsing, so we
318  // have to manually search for a --driver-mode=cl argument the hard way.
319  // Finally, our -cc1 tools don't care which tokenization mode we use because
320  // response files written by clang will tokenize the same way in either mode.
321  bool ClangCLMode = false;
322  if (StringRef(TargetAndMode.DriverMode).equals("--driver-mode=cl") || llvm::find_if(argv, [](const char *F) {
323  return F && strcmp(F, "--driver-mode=cl") == 0;
324  }) != argv.end()) {
325  ClangCLMode = true;
326  }
327  enum { Default, POSIX, Windows } RSPQuoting = Default;
328  for (const char *F : argv) {
329  if (strcmp(F, "--rsp-quoting=posix") == 0) {
330  RSPQuoting = POSIX;
331  } else if (strcmp(F, "--rsp-quoting=windows") == 0) {
332  RSPQuoting = Windows;
333  }
334  }
335 
336  // Determines whether we want nullptr markers in argv to indicate response
337  // files end-of-lines. We only use this for the /LINK driver argument with
338  // clang-cl.exe on Windows.
339  bool MarkEOLs = ClangCLMode;
340 
341  llvm::cl::TokenizerCallback Tokenizer;
342  if (RSPQuoting == Windows || (RSPQuoting == Default && ClangCLMode)) {
343  Tokenizer = &llvm::cl::TokenizeWindowsCommandLine;
344  } else {
345  Tokenizer = &llvm::cl::TokenizeGNUCommandLine;
346  }
347 
348  if (MarkEOLs && argv.size() > 1 && StringRef(argv[1]).startswith("-cc1")) {
349  MarkEOLs = false;
350  }
351  llvm::cl::ExpandResponseFiles(Saver, Tokenizer, argv, MarkEOLs);
352 
353  // Handle -cc1 integrated tools, even if -cc1 was expanded from a response
354  // file.
355  auto FirstArg = std::find_if(argv.begin() + 1, argv.end(), [](const char *A) { return A != nullptr; });
356  if (FirstArg != argv.end() && StringRef(*FirstArg).startswith("-cc1")) {
357  // If -cc1 came from a response file, remove the EOL sentinels.
358  if (MarkEOLs) {
359  auto newEnd = std::remove(argv.begin(), argv.end(), nullptr);
360  argv.resize((size_t) (newEnd - argv.begin()));
361  }
362  return ExecuteCC1Tool(argv);
363  }
364 
365  // Handle options that need handling before the real command line parsing in
366  // Driver::BuildCompilation()
367  bool CanonicalPrefixes = true;
368  for (size_t i = 1, size = argv.size(); i < size; ++i) {
369  // Skip end-of-line response file markers
370  if (argv[i] == nullptr) {
371  continue;
372  }
373  if (StringRef(argv[i]) == "-no-canonical-prefixes") {
374  CanonicalPrefixes = false;
375  break;
376  }
377  }
378 
379  // Handle CL and _CL_ which permits additional command line options to be
380  // prepended or appended.
381  if (ClangCLMode) {
382  // Arguments in "CL" are prepended.
383  llvm::Optional<str> OptCL = llvm::sys::Process::GetEnv("CL");
384  if (OptCL.hasValue()) {
385  SmallVector<const char *, 8> PrependedOpts;
386  getCLEnvVarOptions(OptCL.getValue(), Saver, PrependedOpts);
387 
388  // Insert right after the program name to prepend to the argument list.
389  argv.insert(argv.begin() + 1, PrependedOpts.begin(), PrependedOpts.end());
390  }
391  // Arguments in "_CL_" are appended.
392  llvm::Optional<str> Opt_CL_ = llvm::sys::Process::GetEnv("_CL_");
393  if (Opt_CL_.hasValue()) {
394  SmallVector<const char *, 8> AppendedOpts;
395  getCLEnvVarOptions(Opt_CL_.getValue(), Saver, AppendedOpts);
396 
397  // Insert at the end of the argument list to append.
398  argv.append(AppendedOpts.begin(), AppendedOpts.end());
399  }
400  }
401 
402  std::set<str> SavedStrings;
403  // Handle CCC_OVERRIDE_OPTIONS, used for editing a command line behind the
404  // scenes.
405  if (const char *OverrideStr = ::getenv("CCC_OVERRIDE_OPTIONS")) {
406  // FIXME: Driver shouldn't take extra initial argument.
407  ApplyQAOverride(argv, OverrideStr, SavedStrings);
408  }
409 
410  str Path = GetExecutablePath(argv[0], CanonicalPrefixes);
411 
412  // Whether the cc1 tool should be called inside the current process, or if we
413  // should spawn a new clang subprocess (old behavior).
414  // Not having an additional process saves some execution time of Windows,
415  // and makes debugging and profiling easier.
416  bool UseNewCC1Process;
417 
418  IntrusiveRefCntPtr<DiagnosticOptions> DiagOpts = CreateAndPopulateDiagOpts(argv, UseNewCC1Process);
419 
420  TextDiagnosticPrinter *DiagClient = new TextDiagnosticPrinter(llvm::errs(), &*DiagOpts);
421  FixupDiagPrefixExeName(DiagClient, Path);
422 
423  IntrusiveRefCntPtr<DiagnosticIDs> DiagID(new DiagnosticIDs());
424 
425  DiagnosticsEngine Diags(DiagID, &*DiagOpts, DiagClient);
426 
427  if (!DiagOpts->DiagnosticSerializationFile.empty()) {
428  auto SerializedConsumer =
429  clang::serialized_diags::create(DiagOpts->DiagnosticSerializationFile, &*DiagOpts, /*MergeChildRecords=*/true);
430  Diags.setClient(new ChainedDiagnosticConsumer(Diags.takeClient(), std::move(SerializedConsumer)));
431  }
432 
433  ProcessWarningOptions(Diags, *DiagOpts, /*ReportDiags=*/false);
434 
435  Driver TheDriver(Path, llvm::sys::getDefaultTargetTriple(), Diags);
436  SetInstallDir(argv, TheDriver, CanonicalPrefixes);
437  TheDriver.setTargetAndMode(TargetAndMode);
438 
439  insertTargetAndModeArgs(TargetAndMode, argv, SavedStrings);
440 
441  SetBackdoorDriverOutputsFromEnvVars(TheDriver);
442 
443  if (!UseNewCC1Process) {
444  TheDriver.CC1Main = &ExecuteCC1Tool;
445  // Ensure the CC1Command actually catches cc1 crashes
446  llvm::CrashRecoveryContext::Enable();
447  }
448 
449  std::unique_ptr<Compilation> C(TheDriver.BuildCompilation(argv));
450  int Res = 1;
451  if (C && !C->containsError()) {
452  SmallVector<std::pair<int, const Command *>, 4> FailingCommands;
453  Res = TheDriver.ExecuteCompilation(*C, FailingCommands);
454 
455  // Force a crash to test the diagnostics.
456  if (TheDriver.GenReproducer) {
457  Diags.Report(diag::err_drv_force_crash) << !::getenv("FORCE_CLANG_DIAGNOSTICS_CRASH");
458 
459  // Pretend that every command failed.
460  FailingCommands.clear();
461  for (const auto &J : C->getJobs()) {
462  if (const Command *C = dyn_cast<Command>(&J)) {
463  FailingCommands.push_back(std::make_pair(-1, C));
464  }
465  }
466  }
467 
468  for (const auto &P : FailingCommands) {
469  int CommandRes = P.first;
470  const Command *FailingCommand = P.second;
471  if (!Res) {
472  Res = CommandRes;
473  }
474 
475  // If result status is < 0, then the driver command signalled an error.
476  // If result status is 70, then the driver command reported a fatal error.
477  // On Windows, abort will return an exit code of 3. In these cases,
478  // generate additional diagnostic information if possible.
479  bool DiagnoseCrash = CommandRes < 0 || CommandRes == 70;
480  #ifdef _WIN32
481  DiagnoseCrash |= CommandRes == 3;
482  #endif
483  if (DiagnoseCrash) {
484  TheDriver.generateCompilationDiagnostics(*C, *FailingCommand);
485  break;
486  }
487  }
488  }
489 
490  Diags.getClient()->finish();
491 
492  // If any timers were active but haven't been destroyed yet, print their
493  // results now. This happens in -disable-free mode.
494  llvm::TimerGroup::printAll(llvm::errs());
495  llvm::TimerGroup::clearAll();
496 
497  #ifdef _WIN32
498  // Exit status should not be negative on Win32, unless abnormal termination.
499  // Once abnormal termination was caught, negative status should not be
500  // propagated.
501  if (Res < 0)
502  Res = 1;
503  #endif
504 
505  // If we have multiple failing commands, we return the result of the first
506  // failing command.
507  return Res;
508 }

516 int clang_compile(vector<const char *> input_files, TanCompilation *config) {
517  vector<const char *> args;
518  args.reserve(input_files.size() + 2 * config->n_import_dirs + 1);
519  args.push_back("clang");
520  args.insert(args.end(), input_files.begin(), input_files.end());
521  for (size_t i = 0; i < config->n_import_dirs; ++i) {
522  args.push_back("-I");
523  args.push_back(config->import_dirs[i]);
524  }
525  args.push_back("-c");
526 
527  llvm::IntrusiveRefCntPtr<clang::DiagnosticIDs> diag_id(new clang::DiagnosticIDs());
528  auto diag_options = new clang::DiagnosticOptions();
529  clang::DiagnosticsEngine
530  diag_engine(diag_id, diag_options, new clang::TextDiagnosticPrinter(llvm::errs(), diag_options));
531  auto clang_or_err = llvm::sys::findProgramByName("clang", {__STR__(LLVM_BIN_DIR)});
532  if (!clang_or_err) {
533  std::cerr << "Cannot find clang executable: " << clang_or_err.getError() << "\n";
534  return 1;
535  }
536  clang::driver::Driver driver(clang_or_err.get(), llvm::sys::getDefaultTargetTriple(), diag_engine);
537 
538  auto *compilation = driver.BuildCompilation(args);
539  SmallVector<std::pair<int, const Command *>, 0> failing_commands;
540  if (compilation) { return driver.ExecuteCompilation(*compilation, failing_commands); }
541  return 1;
542 }
