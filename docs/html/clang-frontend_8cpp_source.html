34 static void insertTargetAndModeArgs(const ParsedClangName &NameParts,
35  SmallVectorImpl<const char *> &ArgVector,
36  std::set<str> &SavedStrings) {
37  // Put target and mode arguments at the start of argument list so that
38  // arguments specified in command line could override them. Avoid putting
39  // them at index 0, as an option like '-cc1' must remain the first.
40  int InsertionPoint = 0;
41  if (ArgVector.size() > 0) { ++InsertionPoint; }
42 
43  if (NameParts.DriverMode) {
44  // Add the mode flag to the arguments.
45  ArgVector.insert(ArgVector.begin() + InsertionPoint, GetStableCStr(SavedStrings, NameParts.DriverMode));
46  }
47 
48  if (NameParts.TargetIsValid) {
49  const char *arr[] = {"-target", GetStableCStr(SavedStrings, NameParts.TargetPrefix)};
50  ArgVector.insert(ArgVector.begin() + InsertionPoint, std::begin(arr), std::end(arr));
51  }
52 }

74 static void FixupDiagPrefixExeName(TextDiagnosticPrinter *DiagClient, const str &Path) {
75  // If the clang binary happens to be named cl.exe for compatibility reasons,

81  DiagClient->setPrefix(ExeBasename);

96 int main0(int argc_, const char **argv_) {
97  noteBottomOfStack();
98  llvm::InitLLVM X(argc_, argv_);
99  SmallVector<const char *, 256> argv(argv_, argv_ + argc_);
100  if (llvm::sys::Process::FixupStandardFileDescriptors()) { return 1; }
101 
102  llvm::InitializeAllTargets();
103  auto TargetAndMode = ToolChain::getTargetAndModeFromProgramName(argv[0]);
104 
105  llvm::BumpPtrAllocator A;
106  llvm::StringSaver Saver(A);
107 
108  llvm::cl::TokenizerCallback Tokenizer = &llvm::cl::TokenizeGNUCommandLine;
109  llvm::cl::ExpandResponseFiles(Saver, Tokenizer, argv, false);
110 
111  std::set<str> SavedStrings;
112  str Path = GetExecutablePath(argv[0]);
113  IntrusiveRefCntPtr<DiagnosticOptions> DiagOpts = CreateAndPopulateDiagOpts(argv);
114  TextDiagnosticPrinter *DiagClient = new TextDiagnosticPrinter(llvm::errs(), &*DiagOpts);
115  FixupDiagPrefixExeName(DiagClient, Path);
116  IntrusiveRefCntPtr<DiagnosticIDs> DiagID(new DiagnosticIDs());
117  DiagnosticsEngine Diags(DiagID, &*DiagOpts, DiagClient);
118 
119  if (!DiagOpts->DiagnosticSerializationFile.empty()) {
120  auto SerializedConsumer = clang::serialized_diags::create(DiagOpts->DiagnosticSerializationFile, &*DiagOpts, true);
121  Diags.setClient(new ChainedDiagnosticConsumer(Diags.takeClient(), std::move(SerializedConsumer)));
122  }
123 
124  ProcessWarningOptions(Diags, *DiagOpts, false);
125 
126  Driver TheDriver(Path, llvm::sys::getDefaultTargetTriple(), Diags);
127  TheDriver.setTargetAndMode(TargetAndMode);
128  insertTargetAndModeArgs(TargetAndMode, argv, SavedStrings);
129  SetBackdoorDriverOutputsFromEnvVars(TheDriver);
130 
131  std::unique_ptr<Compilation> C(TheDriver.BuildCompilation(argv));
132  int Res = 1;
133  if (C && !C->containsError()) {
134  SmallVector<std::pair<int, const Command *>, 4> FailingCommands;
135  Res = TheDriver.ExecuteCompilation(*C, FailingCommands);
136 
137  for (const auto &P : FailingCommands) {
138  int CommandRes = P.first;
139  const Command *FailingCommand = P.second;
140  if (!Res) {
141  Res = CommandRes;
142  }
143 
144  // If result status is < 0, then the driver command signalled an error.
145  // If result status is 70, then the driver command reported a fatal error.
146  // On Windows, abort will return an exit code of 3. In these cases,
147  // generate additional diagnostic information if possible.
148  bool DiagnoseCrash = CommandRes < 0 || CommandRes == 70;
149  #ifdef _WIN32
150  DiagnoseCrash |= CommandRes == 3;
151  #endif
152  if (DiagnoseCrash) {
153  TheDriver.generateCompilationDiagnostics(*C, *FailingCommand);
154  break;
155  }
156  }
157  }
158  Diags.getClient()->finish();
159 
160  #ifdef _WIN32
161  // Exit status should not be negative on Win32, unless abnormal termination.
162  // Once abnormal termination was caught, negative status should not be
163  // propagated.
164  if (Res < 0)
165  Res = 1;
166  #endif
167 
168  /// if we have multiple failing commands, we return the result of the first failing command
169  return Res;
170 }

178 int clang_compile(vector<str> input_files, TanCompilation *config) {
179  // TODO: Optimization level
180  vector<const char *> args;
181  size_t n_import = config->import_dirs.size();
182  args.reserve(input_files.size() + 2 * +1);
183  args.push_back("clang");
184  std::for_each(input_files.begin(), input_files.end(), [&args](const auto &s) { args.push_back(s.c_str()); });
185  for (size_t i = 0; i < n_import; ++i) {
186  args.push_back("-I");
187  args.push_back(config->import_dirs[i].c_str());
188  }
189  args.push_back("-c");
190 
191  llvm::IntrusiveRefCntPtr<clang::DiagnosticIDs> diag_id(new clang::DiagnosticIDs());
192  auto diag_options = new clang::DiagnosticOptions();
193  clang::DiagnosticsEngine
194  diag_engine(diag_id, diag_options, new clang::TextDiagnosticPrinter(llvm::errs(), diag_options));
195  clang::driver::Driver driver(GetExecutablePath("clang"), llvm::sys::getDefaultTargetTriple(), diag_engine);
196 
197  auto *compilation = driver.BuildCompilation(args);
198  SmallVector<std::pair<int, const Command *>, 0> failing_commands;
199  if (compilation) { return driver.ExecuteCompilation(*compilation, failing_commands); }
200  return 1;
201 }
