24 /**
25  * @brief Zero overhead unique identifier.
26  *
27  * A hashed string is a compile-time tool that allows users to use
28  * human-readable identifers in the codebase while using their numeric
29  * counterparts at runtime.<br/>
30  * Because of that, a hashed string can also be used in constant expressions if
31  * required.
32  *
33  * @tparam Char Character type.
34  */
35 template<typename Char> class basic_hashed_string {
36  using traits_type = fnv1a_traits<uint32_t>;
37 
38  struct const_wrapper {
39  // non-explicit constructor on purpose
40  constexpr const_wrapper(const Char *curr) noexcept: str{curr} {}
41  const Char *str;
42  };
43 
44  // Fowler–Noll–Vo hash function v. 1a - the good
45  static constexpr uint32_t helper(const Char *curr) noexcept {
46  auto value = traits_type::offset;
47 
48  while (*curr != 0) {
49  value = (value ^ static_cast<traits_type::type>(*(curr++))) * traits_type::prime;
50  }
51 
52  return value;
53  }

56  /*! @brief Character type. */
57  using value_type = Char;
58  /*! @brief Unsigned integer type. */
59  using hash_type = uint32_t;

61  /**
62  * @brief Returns directly the numeric representation of a string.
63  *
64  * Forcing template resolution avoids implicit conversions. An
65  * human-readable identifier can be anything but a plain, old bunch of
66  * characters.<br/>
67  * Example of use:
68  * @code{.cpp}
69  * const auto value = basic_hashed_string<char>::to_value("my.png");
70  * @endcode
71  *
72  * @tparam N Number of characters of the identifier.
73  * @param str Human-readable identifer.
74  * @return The numeric representation of the string.
75  */
76  template<std::size_t N> static constexpr hash_type value(const value_type (&str)[N]) noexcept {
77  return helper(str);
78  }

89  /**
90  * @brief Returns directly the numeric representation of a string view.
91  * @param str Human-readable identifer.
92  * @param size Length of the string to hash.
93  * @return The numeric representation of the string.
94  */
95  static hash_type value(const value_type *str, std::size_t size) noexcept {
96  uint32_t partial{traits_type::offset};
97  while (size--) { partial = (partial ^ (str++)[0]) * traits_type::prime; }
98  return partial;
99  }
