44Type *TypeCheck::ImplicitTypePromote(Type *t1, Type *t2) {
45 TAN_ASSERT(t1 && t2);
46
47 if (t1 == t2) {
48 return t1;
49 }
50
51 int s1 = t1->get_size_bits();
52 int s2 = t2->get_size_bits();
53 if (t1->is_int() && t2->is_int()) {
54 bool u1 = t1->is_unsigned();
55 bool u2 = t2->is_unsigned();
56
57 if (u1 == u2) { // rule #1
58 return s1 >= s2 ? t1 : t2;
59 } else {
60 // let t1 be the unsigned, t2 be the signed
61 if (!u1) {
62 std::swap(t1, t2);
63 std::swap(s1, s2);
64 std::swap(u1, u2);
65 }
66
67 if (s2 > s1) { // rule #2
68 return t2;
69 } else if (s1 > s2) { // rule #3
70 return t1;
71 } else {
72 return nullptr;
73 }
74 }
75 } else if (t1->is_float() && t2->is_float()) { // rule #4
76 return s1 >= s2 ? t1 : t2;
77 }
78
79 // rule #5
80 else if (t1->is_float() && t2->is_int()) {
81 return t1;
82 } else if (t1->is_int() && t2->is_float()) {
83 return t2;
84 }
85
86 // # rule 6
87 else if (t1->is_bool() && (t2->is_num() || t2->is_pointer())) {
88 return t2;
89 } else if ((t1->is_num() || t1->is_pointer()) && t2->is_bool()) {
