14static str search_library(const vector<str> &lib_dirs, const str &lib_name) {
15 for (const str &dir : lib_dirs) {
16 vector<fs::path> candidates = {
17 /// possible filenames
18 fs::path(dir) / fs::path(lib_name), //
19 fs::path(dir) / fs::path(lib_name + ".a"), //
20 fs::path(dir) / fs::path(lib_name + ".so"), //
21 fs::path(dir) / fs::path("lib" + lib_name + ".a"), //
22 fs::path(dir) / fs::path("lib" + lib_name + ".so"), //
23 };
24 for (const auto &p : candidates) {
25 if (fs::exists(p)) {
26 return p.string();
27 }
28 }
29 }
30 return "";
31}

67 }

70 for (size_t i = 0; i < n_link_files; ++i) {
71 linker.add_flag("-l" + std::string(config->link_files[i]));
72 }
73 linker.add_flag(opt_level_to_string(config->opt_level));
74 return linker.link();

77bool compile_files(vector<str> input_paths, TanCompilation *config) {
78 bool print_ir_code = config->verbose >= 1;
79 bool print_ast = config->verbose >= 2;
80
81 /// input files
82 size_t n_files = input_paths.size();
83 vector<str> files;
84 files.reserve(n_files);
85 vector<str> obj_files;
86 obj_files.reserve(n_files);
87 for (size_t i = 0; i < n_files; ++i) {
88 if (fs::path(input_paths[i]).extension() == ".tan") {
89 files.push_back(input_paths[i]);
90 } else if (fs::path(input_paths[i]).extension() == ".o") {
91 obj_files.push_back(input_paths[i]);
92 } else {
93 std::cerr << "Unknown file extension: " << input_paths[i] << "\n";
94 return false;
95 }
96 }
97 n_files = files.size();
98 /// config
99 Compiler::compile_config = *config;
100 /// import dirs
101 size_t n_import = config->import_dirs.size();
102 Compiler::import_dirs.reserve(n_import);
103 Compiler::import_dirs.insert(Compiler::import_dirs.begin(), config->import_dirs.begin(), config->import_dirs.end());
104
105 // Parse
106 Compiler compiler(files);
107 compiler.parse();
108 if (print_ast) {
109 compiler.dump_ast();
110 }
111
112 // Semantic analysis
113 compiler.analyze();
114
115 // Code generation
116 vector<str> out_files(n_files);
117 size_t i = 0;
118 for (auto *cu : compiler.get_compilation_units()) {
119 // prepare filename for linking
120 str ofile = out_files[i] = fs::path(cu->filename() + ".o").filename().string();
121 std::cout << fmt::format("Compiling TAN file: {}\n", ofile);
122
123 compiler.codegen(cu, print_ir_code);
124 compiler.emit_object(cu, ofile);
125
126 ++i;
127 }
128
129 // Link
130 out_files.insert(out_files.begin(), obj_files.begin(), obj_files.end());
131 if (config->type != OBJ) {
132 bool ret = _link(out_files, config);
133 if (!ret) {
134 std::cerr << "Error linking files\n";
135 }
136 return ret;
137 } else {
138 return true;
139 }
140}
