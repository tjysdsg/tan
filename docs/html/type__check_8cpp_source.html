18void TypeCheck::run_impl(Package *p) {
19 push_scope(p);
20
21 // check unresolved symbols
22 auto [unresolved_symbols, error_node] = p->top_level_symbol_dependency.topological_sort();
23 if (error_node.has_value()) {
24 error(ErrorType::COMPILE_ERROR, error_node.value(), "Cyclic dependency detected");
25 }
26
27 for (auto *c : unresolved_symbols.value()) {
28 visit(c);
29 }
30
31 // check remaining ones
32 std::set<ASTBase *> s(unresolved_symbols.value().begin(), unresolved_symbols.value().end());
33 for (auto *c : p->get_children()) {
34 if (!s.contains(c))
35 visit(c);
36 }
37
38 pop_scope();
39}

41FunctionDecl *TypeCheck::search_function_callee(FunctionCall *p) {
42 const str &name = p->get_name();
43 const vector<Expr *> &args = p->_args;
44
45 FunctionDecl *candidate = top_ctx()->get_func_decl(name);
46 if (!candidate) {
47 error(ErrorType::TYPE_ERROR, p, fmt::format("Unknown function call: {}", name));
48 }
49
50 size_t n = candidate->get_n_args();
51 if (n != args.size()) {
52 error(ErrorType::SEMANTIC_ERROR, p,
53 fmt::format("Incorrect number of arguments: expect {} but found {}", candidate->get_n_args(), n));
54 }
55
56 auto *func_type = pcast<FunctionType>(candidate->get_type());
57
58 // Check if argument types match (return type not checked)
59 // Allow implicit cast from actual arguments to expected arguments
60 for (size_t i = 0; i < n; ++i) {
61 auto *actual_type = args[i]->get_type();
62 auto *expected_type = func_type->get_arg_types()[i];
63
64 if (actual_type != expected_type) {
65 if (!CanImplicitlyConvert(actual_type, expected_type)) {
66 error(ErrorType::TYPE_ERROR, p,
67 fmt::format("Cannot implicitly convert the type of argument {}: expect {} but found {}", i + 1,
68 actual_type->get_typename(), expected_type->get_typename()));
69 }
70 }
71 }
72
73 return candidate;
74}

76Type *TypeCheck::resolve_type_ref(Type *p, ASTBase *node) {
77 TAN_ASSERT(p->is_ref());
78 Type *ret = p;
79
80 const str &referred_name = p->get_typename();

83 ret = decl->get_type();
84 } else {
85 error(ErrorType::TYPE_ERROR, node, fmt::format("Unknown type {}", referred_name));
86 }
87
88 TAN_ASSERT(ret);
89 return ret;

92Type *TypeCheck::resolve_type(Type *p, ASTBase *node) {
93 TAN_ASSERT(p);
94
95 Type *ret = p;
96 if (p->is_ref()) {
97 ret = resolve_type_ref(p, node);
98 } else if (p->is_pointer()) {
99 auto *pointee = pcast<PointerType>(p)->get_pointee();
100
101 TAN_ASSERT(pointee);
102 if (pointee->is_ref()) {
103 pointee = resolve_type_ref(pointee, node);
104 ret = Type::GetPointerType(pointee);
105 }
106 }
107
108 TAN_ASSERT(ret);
109 return ret;
110}

112void TypeCheck::analyze_func_decl_prototype(ASTBase *_p) {
113 auto *p = pcast<FunctionDecl>(_p);
114
115 push_scope(p);
116
117 /// update return type
118 auto *func_type = pcast<FunctionType>(p->get_type());
119 auto *ret_type = resolve_type(func_type->get_return_type(), p);
120 func_type->set_return_type(ret_type);
121
122 /// type_check_ast args
123 size_t n = p->get_n_args();
124 const auto &arg_decls = p->get_arg_decls();
125 vector<Type *> arg_types(n, nullptr);
126 for (size_t i = 0; i < n; ++i) {
127 visit(arg_decls[i]); /// args will be added to the scope here
128 arg_types[i] = arg_decls[i]->get_type();
129 TAN_ASSERT(arg_types[i]->is_canonical());
130 }
131 func_type->set_arg_types(arg_types); /// update arg types
132
133 pop_scope();
134}

148void TypeCheck::analyze_function_call(FunctionCall *p, bool include_intrinsics) {
149 for (const auto &a : p->_args) {
150 visit(a);
151 }
152
153 FunctionDecl *callee = search_function_callee(p);
154 if (include_intrinsics || !callee->is_intrinsic()) {
155 p->_callee = callee;
156 } else {
157 error(ErrorType::UNKNOWN_SYMBOL, p,
158 fmt::format("Unknown function call. Maybe use @{} if you want to call this intrinsic?", p->get_name()));
159 }
160
161 auto *func_type = pcast<FunctionType>(callee->get_type());
162 p->set_type(func_type->get_return_type());
163}

165void TypeCheck::analyze_intrinsic_func_call(Intrinsic *p, FunctionCall *func_call) {
166 auto *void_type = Type::GetVoidType();
167 switch (p->get_intrinsic_type()) {
168 case IntrinsicType::STACK_TRACE:
169 func_call->set_name(Intrinsic::STACK_TRACE_FUNCTION_REAL_NAME);
170 [[fallthrough]];
171 case IntrinsicType::ABORT:
172 analyze_function_call(func_call, true);
173 p->set_type(void_type);
174 break;
175 case IntrinsicType::GET_DECL: {
176 if (func_call->get_n_args() != 1) {
177 error(ErrorType::SEMANTIC_ERROR, func_call, "Expect the number of args to be 1");
178 }
179
180 auto *target = func_call->get_arg(0);
181 if (target->get_node_type() != ASTNodeType::ID) {
182 error(ErrorType::TYPE_ERROR, target,
183 fmt::format("Expect an identifier as the operand, but got {}",
184 ASTBase::ASTTypeNames[target->get_node_type()]));
185 }
186
187 visit(target);
188
189 auto *id = pcast<Identifier>(target);
190 if (id->get_id_type() != IdentifierType::ID_VAR_REF) {
191 error(ErrorType::TYPE_ERROR, id, fmt::format("Expect a value but got type"));
192 }
193
194 auto *decl = id->get_var_ref()->get_referred();
195 auto *source_str = Literal::CreateStringLiteral(p->src(), decl->src()->get_source_code(decl->start(), decl->end()));
196
197 // FEATURE: Return AST?
198 p->set_sub(source_str);
199 p->set_type(source_str->get_type());
200 break;
201 }
202 case IntrinsicType::COMP_PRINT: {
203 p->set_type(void_type);
204
205 // FEATURE: print with var args
206 auto args = func_call->_args;
207 if (args.size() != 1 || args[0]->get_node_type() != ASTNodeType::STRING_LITERAL) {
208 error(ErrorType::TYPE_ERROR, p, "Invalid call to compprint, one argument with type 'str' required");
209 }
210
211 str msg = pcast<StringLiteral>(args[0])->get_value();
212 std::cout << fmt::format("Message ({}): {}\n", p->src()->get_src_location_str(p->start()), msg);
213 break;
214 }
215 default:
216 TAN_ASSERT(false);
217 }
218}

221void TypeCheck::analyze_member_func_call(MemberAccess *p, Expr *lhs, FunctionCall *rhs) {
222 if (!lhs->is_lvalue() && !lhs->get_type()->is_pointer()) {
223 error(ErrorType::TYPE_ERROR, p, "Invalid member function call");
224 }
225
226 // insert the address of the struct instance as the first parameter
227 if (lhs->is_lvalue() && !lhs->get_type()->is_pointer()) {
228 Expr *tmp = UnaryOperator::Create(UnaryOpKind::ADDRESS_OF, lhs->src(), lhs);
229 visit(tmp);
230 rhs->_args.insert(rhs->_args.begin(), tmp);
231 } else {
232 rhs->_args.insert(rhs->_args.begin(), lhs);
233 }
234
235 visit(rhs);
236 p->set_type(rhs->get_type());
237}

240void TypeCheck::analyze_bracket_access(MemberAccess *p, Expr *lhs, Expr *rhs) {
241 visit(rhs);
242
243 if (!lhs->is_lvalue()) {
244 error(ErrorType::TYPE_ERROR, p, "Expect lhs to be an lvalue");
245 }
246
247 auto *lhs_type = lhs->get_type();
248 if (!(lhs_type->is_pointer() || lhs_type->is_array() || lhs_type->is_string())) {
249 error(ErrorType::TYPE_ERROR, p, "Expect a type that supports bracket access");
250 }
251 if (!rhs->get_type()->is_int()) {
252 error(ErrorType::TYPE_ERROR, rhs, "Expect an integer");
253 }
254
255 Type *sub_type = nullptr;
256 if (lhs_type->is_pointer()) {
257 sub_type = pcast<PointerType>(lhs_type)->get_pointee();
258 } else if (lhs_type->is_array()) {
259 auto *array_type = pcast<ArrayType>(lhs_type);
260 sub_type = array_type->get_element_type();
261 /// check if array index is out-of-bound
262 if (rhs->get_node_type() == ASTNodeType::INTEGER_LITERAL) {
263 uint64_t size = pcast<IntegerLiteral>(rhs)->get_value();
264 if (lhs->get_type()->is_array() && (int)size >= array_type->array_size()) {
265 error(ErrorType::TYPE_ERROR, p,
266 fmt::format("Index {} out of bound, the array size is {}", std::to_string(size),
267 std::to_string(array_type->array_size())));
268 }
269 }
270 } else if (lhs_type->is_string()) {
271 sub_type = Type::GetCharType();
272 }
273
274 p->set_type(sub_type);
275}

278void TypeCheck::analyze_member_access_member_variable(MemberAccess *p, Expr *lhs, Expr *rhs) {
279 str m_name = pcast<Identifier>(rhs)->get_name();
280 Type *struct_ty = nullptr;
281 /// auto dereference pointers
282 if (lhs->get_type()->is_pointer()) {
283 struct_ty = pcast<PointerType>(lhs->get_type())->get_pointee();
284 } else {
285 struct_ty = lhs->get_type();
286 }
287
288 struct_ty = resolve_type(struct_ty, lhs);
289 if (!struct_ty->is_struct()) {
290 error(ErrorType::TYPE_ERROR, lhs, "Expect a struct type");
291 }
292
293 auto *struct_decl = pcast<StructDecl>(search_decl_in_scopes(struct_ty->get_typename()));
294 p->_access_idx = struct_decl->get_struct_member_index(m_name);
295 if (p->_access_idx == -1) {
296 error(ErrorType::UNKNOWN_SYMBOL, p,
297 fmt::format("Cannot find member variable '{}' of struct '{}'", m_name, struct_decl->get_name()));
298 }
299 auto *ty = struct_decl->get_struct_member_ty(p->_access_idx);
300 p->set_type(resolve_type(ty, p));
301}

379DEFINE_AST_VISITOR_IMPL(TypeCheck, BinaryOperator) {
380 Expr *lhs = p->get_lhs();
381 Expr *rhs = p->get_rhs();
382
383 if (p->get_op() == BinaryOpKind::MEMBER_ACCESS) {
384 CALL_AST_VISITOR(MemberAccess, p);
385 return;
386 }
387
388 visit(lhs);
389 visit(rhs);
390
391 switch (p->get_op()) {
392 case BinaryOpKind::SUM:
393 case BinaryOpKind::SUBTRACT:
394 case BinaryOpKind::MULTIPLY:
395 case BinaryOpKind::DIVIDE:
396 case BinaryOpKind::BAND:
397 case BinaryOpKind::BOR:
398 case BinaryOpKind::MOD: {
399 p->set_type(auto_promote_bop_operand_types(p));
400 break;
401 }
402 case BinaryOpKind::LAND:
403 case BinaryOpKind::LOR:
404 case BinaryOpKind::XOR: {
405 // check if both operators are bool
406 auto *bool_type = PrimitiveType::GetBoolType();
407 p->set_lhs(create_implicit_conversion(lhs, bool_type));
408 p->set_rhs(create_implicit_conversion(rhs, bool_type));
409 p->set_type(bool_type);
410 break;
411 }
412 case BinaryOpKind::GT:
413 case BinaryOpKind::GE:
414 case BinaryOpKind::LT:
415 case BinaryOpKind::LE:
416 case BinaryOpKind::EQ:
417 case BinaryOpKind::NE:
418 auto_promote_bop_operand_types(p);
419 p->set_type(PrimitiveType::GetBoolType());
420 break;
421 default:
422 TAN_ASSERT(false);
423 }

426DEFINE_AST_VISITOR_IMPL(TypeCheck, UnaryOperator) {
427 auto *rhs = p->get_rhs();
428 visit(rhs);
429
430 auto *rhs_type = rhs->get_type();
431 switch (p->get_op()) {
432 case UnaryOpKind::LNOT:
433 rhs = create_implicit_conversion(rhs, Type::GetBoolType());
434 p->set_rhs(rhs);
435 p->set_type(PrimitiveType::GetBoolType());
436 break;
437 case UnaryOpKind::BNOT:
438 if (!rhs_type->is_int()) {
439 error(ErrorType::TYPE_ERROR, rhs, "Expect an integer type");
440 }
441 p->set_type(rhs_type);
442 break;
443 case UnaryOpKind::ADDRESS_OF:
444 p->set_type(Type::GetPointerType(rhs_type));
445 break;
446 case UnaryOpKind::PTR_DEREF:
447 if (!rhs_type->is_pointer()) {
448 error(ErrorType::TYPE_ERROR, rhs, "Expect a pointer type");
449 }
450 TAN_ASSERT(rhs->is_lvalue());
451 p->set_lvalue(true);
452 p->set_type(pcast<PointerType>(rhs_type)->get_pointee());
453 break;
454 case UnaryOpKind::PLUS:
455 case UnaryOpKind::MINUS: /// unary plus/minus
456 if (!rhs_type->is_num()) {
457 error(ErrorType::TYPE_ERROR, rhs, "Expect a numerical type");
458 }
459 p->set_type(rhs_type);
460 break;
461 default:
462 TAN_ASSERT(false);
463 }
464}

472DEFINE_AST_VISITOR_IMPL(TypeCheck, Assignment) {
473 Expr *rhs = p->get_rhs();
474 visit(rhs);
475
476 auto *lhs = p->get_lhs();
477 Type *lhs_type = nullptr;
478 if (lhs->get_node_type() == ASTNodeType::VAR_DECL) {
479 auto *var_decl = pcast<VarDecl>(lhs);
480
481 // deduce type of variable declaration
482 if (!var_decl->get_type()) {
483 var_decl->set_type(rhs->get_type());
484 }
485
486 visit(lhs);
487 lhs_type = var_decl->get_type();
488 } else {
489 visit(lhs);
490
491 switch (lhs->get_node_type()) {
492 case ASTNodeType::ID: {
493 auto *id = pcast<Identifier>(lhs);
494 if (id->get_id_type() != IdentifierType::ID_VAR_REF) {
495 error(ErrorType::TYPE_ERROR, lhs, "Can only assign value to a variable");
496 }
497 lhs_type = id->get_type();
498 break;
499 }
500 case ASTNodeType::ARG_DECL:
501 case ASTNodeType::BOP_OR_UOP:
502 case ASTNodeType::UOP:
503 case ASTNodeType::BOP:
504 lhs_type = pcast<Expr>(lhs)->get_type();
505 break;
506 default:
507 error(ErrorType::TYPE_ERROR, lhs, "Invalid left-hand operand");
508 }
509 }
510 p->set_type(lhs_type);
511
512 rhs = create_implicit_conversion(rhs, lhs_type);
513 p->set_rhs(rhs);
514 p->set_lvalue(true);
515}

532DEFINE_AST_VISITOR_IMPL(TypeCheck, Intrinsic) {
533 switch (p->get_intrinsic_type()) {
534 case IntrinsicType::LINENO: {
535 auto sub = IntegerLiteral::Create(p->src(), p->src()->get_line(p->start()), true);
536 auto type = PrimitiveType::GetIntegerType(32, true);
537 sub->set_type(type);
538 p->set_type(type);
539 p->set_sub(sub);
540 break;
541 }
542 case IntrinsicType::FILENAME: {
543 auto sub = StringLiteral::Create(p->src(), p->src()->get_filename());
544 auto type = Type::GetStringType();
545 sub->set_type(type);
546 p->set_type(type);
547 p->set_sub(sub);
548 break;
549 }
550 case IntrinsicType::TEST_COMP_ERROR: {
551 auto *tce = pcast<TestCompError>(p->get_sub());
552 if (tce->_caught)
553 return;
554
555 push_scope(p);
556
557 try {
558 visit(tce);
559 } catch (const CompileException &e) {
560 std::cerr << fmt::format("Caught expected compile error: {}\nContinue compilation...\n", e.what());
561 tce->_caught = true;
562 }
563
564 if (!tce->_caught)
565 error(ErrorType::TYPE_ERROR, p, "Expect a compile error");
566
567 pop_scope();
568 break;
569 }
570 case IntrinsicType::NOOP:
571 break;
572 case IntrinsicType::INVALID:
573 TAN_ASSERT(false);
574 break;
575 default: {
576 auto *c = p->get_sub();
577 if (c->get_node_type() == ASTNodeType::FUNC_CALL) {
578 analyze_intrinsic_func_call(p, pcast<FunctionCall>(c));
579 return;
580 }
581
582 break;
583 }
584 }

645DEFINE_AST_VISITOR_IMPL(TypeCheck, StructDecl) {
646 str struct_name = p->get_name();
647 auto *ty = pcast<StructType>(p->get_type());
648 TAN_ASSERT(ty && ty->is_struct());
649 auto members = p->get_member_decls();
650
651 push_scope(p);
652
653 size_t n = members.size();
654 for (size_t i = 0; i < n; ++i) {
655 Expr *m = members[i];
656
657 if (m->get_node_type() == ASTNodeType::VAR_DECL) { // member variable without initial value
658 (*ty)[i] = resolve_type((*ty)[i], m);
659
660 } else if (m->get_node_type() == ASTNodeType::ASSIGN) { // member variable with an initial value
661 auto init_val = pcast<Assignment>(m)->get_rhs();
662 visit(init_val);
663
664 (*ty)[i] = resolve_type((*ty)[i], m);
665
666 if (!init_val->is_comptime_known()) {
667 error(ErrorType::TYPE_ERROR, p, "Initial value of a member variable must be compile-time known");
