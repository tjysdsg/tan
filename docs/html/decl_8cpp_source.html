44 FunctionDecl *FunctionDecl::GetCallee(ASTContext *ctx, const str &name, const vector<Expr *> &args) {
45  FunctionDecl *ret = nullptr;
46  auto func_candidates = ctx->get_functions(name);
47  /// always prefer the function with lowest cost if multiple candidates are callable
48  /// one implicit cast -> +1 cost
49  /// however, if two (or more) functions have the same score, an error is raise (ambiguous call)
50  auto cost = (size_t) -1;
51  for (const auto &f : func_candidates) {
52  size_t n = f->get_n_args();
53  if (n != args.size()) { continue; }
54  bool good = true;
55  size_t c = 0;
56  for (size_t i = 0; i < n; ++i) { /// check every argument (return type not checked)
57  auto actual_arg = args[i];
58  /// allow implicit cast from actual_arg to arg, but not in reverse
59  auto t1 = f->get_arg_type(i);
60  auto t2 = actual_arg->get_type();
61  if (*t1 != *t2) {
62  if (0 != TypeSystem::CanImplicitCast(ctx, t1, t2)) {
63  good = false;
64  break;
65  }
66  ++c;
67  }
68  }
69  if (good) {
70  if (c < cost) {
71  ret = f;
72  cost = c;
73  } else if (c == cost) {
74  throw std::runtime_error("Ambiguous function call: " + name);
75  // FIXME: report_error(ctx, p, "Ambiguous function call: " + name);
76  }
77  }
78  }
79  if (!ret) {
80  throw std::runtime_error("Unknown function call: " + name);
81  // FIXME: report_error(ctx, p, "Unknown function call: " + name);
82  }
83  return ret;
84 }
