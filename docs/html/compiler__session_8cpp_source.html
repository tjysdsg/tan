86 void CompilerSession::emit_object(const str &filename) {
87  _di_builder->finalize(); /// important: do this before any pass
88 
89  /// run function pass on all functions in the current module
90  for (auto &f: *_module) {
91  if (f.getName() == "main") { /// special case for main function
92  // mark as used, prevent LLVM from deleting it
93  llvm::appendToUsed(*_module, {(GlobalValue *) &f});
94  }
95  _fpm->run(f);
96  }
97  _fpm->doFinalization();
98 
99  /// generate object files
100  std::error_code ec;
101  llvm::raw_fd_ostream dest(filename, ec, llvm::sys::fs::OF_None);
102  if (ec) { report_error("Could not open file: " + ec.message()); }
103 
104  auto file_type = llvm::CGFT_ObjectFile;
105  if (_target_machine->addPassesToEmitFile(*_mpm.get(), dest, nullptr, file_type)) {
106  report_error("Target machine can't emit a file of this type");
107  }
108  _mpm->run(*_module);
109  dest.flush();
110 }
