68void Compiler::parse() {
69 for (const str &file : _files) {
70 Reader reader;
71 reader.open(file);
72
73 // tokenization
74 auto tokens = tokenize(&reader);
75
76 auto *sm = new SourceManager(file, tokens);
77 auto *parser = new Parser(sm);
78 auto *ast = parser->parse();
79
80 // register top-level declarations
81 auto intrinsic_funcs = Intrinsic::GetIntrinsicFunctionDeclarations();
82 for (auto *f : intrinsic_funcs) {
83 ast->ctx()->set_function_decl(f);
84 }
85
86 _cu.push_back(new CompilationUnit(ast, sm));
87 }
88}

108vector<str> Compiler::resolve_import(const str &callee_path, const str &import_name) {
109 vector<str> ret{};
110 auto import_path = fs::path(import_name);
111 /// search relative to callee's path
112 {
113 auto p = fs::path(callee_path).parent_path() / import_path;
114 p = p.lexically_normal();
115 if (fs::exists(p)) {
116 ret.push_back(p.string());
117 }
118 }
119 /// search relative to directories in Compiler::import_dirs
120 for (const auto &rel : Compiler::import_dirs) {
121 auto p = fs::path(rel) / import_path;
122 p = p.lexically_normal();
123 if (fs::exists(p)) {
124 ret.push_back(p.string());
125 }
126 }
127 return ret;
128}
