29void Reader::from_string(const str &code) {
30 str line;
31 size_t line_start = 0;
32 str new_line;
33 size_t code_len = code.length();
34 for (size_t c = 0; c < code_len; ++c) {
35 if (code[c] == '\n' || c == code.length() - 1) {
36 if (code[c] == '\n') {
37 line = code.substr(line_start, c - line_start); /// not including trailing '\n'
38 } else {
39 line = code.substr(line_start, c - line_start + 1);
40 }
41 new_line = line;
42 /// delete whitespace at the beginning of the line
43 for (size_t i = 0; i < line.length(); ++i) {
44 if (!isspace(line[i])) { // avoid confusion with isspace in <locale>
45 new_line = line.substr(i);
46 break;
47 }
48 }
49 _lines.push_back(new_line);
50 line_start = c + 1;
51 }
52 }

68str Reader::substr(const Cursor &start, const Cursor &_end) const {
69 TAN_ASSERT(is_cursor_valid(start));
70
71 /// if the end cursor is out of boundary, make it point to EOF
72 Cursor end(_end);
73 if (end.l >= _lines.size() || end.c > _lines[end.l].size()) {
74 end.l = (uint32_t)_lines.size() - 1;
75 end.c = (uint32_t)_lines.back().size();
76 }
77 auto s_row = start.l;
78 auto e_row = end.l;
79 str ret;
80 if (s_row == e_row) {
81 TAN_ASSERT(start.c <= end.c);
82 ret = _lines[s_row].substr(start.c, end.c - start.c);
83 } else {
84 ret += _lines[s_row].substr(start.c) + "\n";
85 for (auto r = s_row + 1; r < e_row; ++r) {
86 ret += _lines[r] + "\n";
87 }
88 if (end.c > 0) {
89 ret += _lines[e_row].substr(0, end.c);
90 }
91 }
92 return ret;
93}
