129void CompilerDriver::run(const vector<str> &files) {
130 // Check if files exist
131 // and separate cxx and tan source files based on their file extensions
132 vector<str> tan_files{};
133 vector<str> cxx_files{};
134 for (size_t i = 0; i < files.size(); ++i) {
135 fs::path f = fs::path(files[i]);
136 str ext = f.extension().string();
137
138 if (!fs::exists(f))
139 Error(ErrorType::FILE_NOT_FOUND, fmt::format("File not found: {}", files[i])).raise();
140
141 bool is_cxx = std::any_of(CXX_EXTS.begin(), CXX_EXTS.end(), [=](const str &e) { return e == ext; });
142 if (is_cxx) {
143 cxx_files.push_back(files[i]);
144 } else if (ext == TAN_EXT) {
145 tan_files.push_back(files[i]);
146 } else {
147 Error(ErrorType::GENERIC_ERROR, fmt::format("Unrecognized source file: {}", files[i])).raise();
148 }
149 }
150
151 // Compiling
152 auto cxx_objs = compile_cxx(cxx_files, _config);
153 auto tan_objs = compile_tan(tan_files);
154
155 // Linking
156 vector<str> obj_files(cxx_objs.size() + tan_objs.size());
157 size_t i = 0;
158 for (const str &o : cxx_objs)
159 obj_files[i++] = o;
160 for (const str &o : tan_objs)
161 obj_files[i++] = o;
162
163 link(obj_files);
164}

176vector<Package *> CompilerDriver::stage1_analysis(vector<Program *> programs) {
177 TAN_ASSERT(!programs.empty());
178
179 // Register all declarations in their local contexts
180 for (auto *p : programs) {
181 RegisterDeclarations rd;
182 rd.run(p);
183 }
184
185 // Organize input files into packages
186 OrganizePackages op;
187 vector<Package *> ps = op.run(programs);
188
189 // Skip packages that are already processed, and check for cyclic dependencies
190 vector<Package *> packages{};
191 for (Package *p : ps) {
192 AnalyzeStatus status = _package_status[p->get_name()];
193 if (status == AnalyzeStatus::None) {
194 packages.push_back(p);
195 } else if (status == AnalyzeStatus::Processing) {
196 // TODO: better error message
197 Error(ErrorType::IMPORT_ERROR, "Cyclic package dependency detected for package: " + p->get_name()).raise();
198 }
199 }
200
201 // Register packages we found BEFORE running semantic analysis,
202 // so that we can search for them during analysis
203 for (auto *p : packages) {
204 register_package(p->get_name(), p);
205 }
206
207 // Scan package imports and find the source files needed
208 for (auto *p : packages) {
209 _package_status[p->get_name()] = AnalyzeStatus::Processing;
210
211 uset<str> import_files{};
212 uset<str> import_names{};
213
214 ScanImports si;
215 auto res = si.run(p);
216 for (const auto &e : res) {
217 import_names.insert(e.first);
218 import_files.insert(e.second.begin(), e.second.end());
219 }
220
221 // Analyze imported files and store results
222 if (!import_files.empty()) {
223 vector<Package *> import_packages = stage1_analysis(parse(vector<str>(import_files.begin(), import_files.end())));
224 for (Package *ip : import_packages) {
225 if (import_names.contains(ip->get_name())) { // import_packages might have some unrelated packages
226 register_package(ip->get_name(), ip);
227 }
228 }
229 }
230
231 _package_status[p->get_name()] = AnalyzeStatus::Done;
232 }
233
234 // Partial type checking
235 for (auto *p : packages) {
236 TypePrecheck tp;
237 tp.run(p);
238 }
239
240 return packages;
241}

243vector<str> CompilerDriver::compile_tan(const vector<str> &files) {
244 bool print_ir_code = _config.verbose >= 1;
245 size_t n_files = files.size();
246 vector<str> ret(n_files);
247
248 // Parse
249 auto programs = parse(files);
250
251 // (Optional): Print AST tree
252 if (_config.verbose >= 2) {
253 for (auto *p : programs) {
254 std::cout << fmt::format("AST Tree of {}:\n{}", p->src()->get_filename(), p->repr());
255 }
256 }
257
258 vector<Package *> packages = stage1_analysis(programs);
259
260 // Full semantic analysis
261 for (auto *p : packages) {
262 TypeCheck analyzer;
263 analyzer.run(p);
264 }
265
266 // Code generation
267 size_t i = 0;
268 for (auto *p : packages) {
269 std::cout << fmt::format("Compiling TAN package: {}\n", p->get_name());
270
271 // IR
272 _target_machine->setOptLevel(tan_to_llvm_opt_level[_config.opt_level]);
273 auto *cg = new CodeGenerator(_target_machine);
274 cg->run(p);
275
276 if (print_ir_code)
277 cg->dump_ir();
278
279 // object file
280 str ofile = ret[i] = fs::path(p->get_name() + ".o").filename().string();
281 cg->emit_to_file(ofile);
282
283 ++i;
284
285 delete cg;
286 }
287
288 for (auto *p : programs) {
289 delete p;
290 }
291 return ret;
292}

294vector<Program *> CompilerDriver::parse(const vector<str> &files) {
295 TAN_ASSERT(!files.empty());
296
297 vector<Program *> ret{};
298
299 for (const str &file : files) {
300 SourceFile *source = new SourceFile();
301 source->open(file);
302
303 // tokenization
304 auto tokens = tokenize(source);
305
306 auto *sm = new TokenizedSourceFile(file, tokens);
307 auto *parser = new Parser(sm);
308 auto *ast = parser->parse();
309
310 // register top-level declarations
311 // TODO: put intrinsics into a dedicated module
312 auto intrinsic_funcs = Intrinsic::GetIntrinsicFunctionDeclarations();
313 for (auto *f : intrinsic_funcs) {
314 ast->ctx()->set_function_decl(f);
315 }
316
317 ret.push_back(ast);
318 }
319
320 return ret;
321}

323vector<str> CompilerDriver::resolve_package_import(const str &callee_path, const str &import_name) {
324 auto import_path = fs::path(import_name);
325
326 // importing using an absolute path
327 if (import_path.is_absolute() && fs::exists(import_path)) {
328 return {import_path}; // no reason to continue
329 }
330
331 vector<str> ret{};
332
333 // search relative to callee's path
334 {
335 auto p = fs::path(callee_path).parent_path() / import_path;
336 p = p.lexically_normal();
337 if (fs::exists(p) || fs::exists(p.replace_extension(".tan"))) {
338 ret.push_back(fs::absolute(p));
339 }
340 }
341
342 // user-defined include dirs
343 for (const auto &rel : CompilerDriver::import_dirs) {
344 auto p = fs::path(rel) / import_path;
345 p = p.lexically_normal();
346 if (fs::exists(p) || fs::exists(p.replace_extension(".tan"))) {
347 ret.push_back(fs::absolute(p));
348 }
349 }
350
351 // TODO: system directories
352 return ret;
353}

355void CompilerDriver::link(const std::vector<str> &files) {
356 if (_config.type == SLIB) { // static
357 // also add files specified by -l option
358 vector<str> all_files(files.begin(), files.end());
359 for (const auto &lib : _config.link_files) {
360 str path = search_library(_config.lib_dirs, lib);
361
362 if (path.empty())
363 Error(ErrorType::LINK_ERROR, fmt::format("Unable to find library: {}", lib)).raise();
364
365 all_files.push_back(path);
366 }
367
368 llvm_ar_create_static_lib(_config.out_file, all_files);
369 return;
370 }
371
372 // shared, obj, or exe
373 using tanlang::Linker;
374 Linker linker;
375 linker.add_files(files);
376 linker.add_flag("-o" + str(_config.out_file));
377 if (_config.type == EXE) {
378 linker.add_flags({"-fPIE"});
379 } else if (_config.type == DLIB) {
380 linker.add_flags({"-shared"});
381 }
382
383 // -L
384 size_t n_lib_dirs = _config.lib_dirs.size();
385 for (size_t i = 0; i < n_lib_dirs; ++i) {
386 auto p = fs::absolute(fs::path(_config.lib_dirs[i]));
387 linker.add_flag("-L" + p.string());
388 linker.add_flag("-Wl,-rpath," + p.string());
389 }
390
391 // -l
392 size_t n_link_files = _config.link_files.size();
393 for (size_t i = 0; i < n_link_files; ++i) {
394 linker.add_flag("-l" + std::string(_config.link_files[i]));
395 }
396 linker.add_flag(opt_level_to_string(_config.opt_level));
397
398 if (!linker.link())
399 Error(ErrorType::LINK_ERROR, "Failed linking").raise();
400}
