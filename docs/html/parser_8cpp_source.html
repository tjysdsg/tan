100  ASTBase *peek() {
101  if (eof(_curr)) { return nullptr; }
102  Token *token = at(_curr);
103  /// skip comments
104  while (token && token->get_type() == TokenType::COMMENTS) {
105  _curr.offset_by(1);
106  if (eof(_curr)) { return nullptr; }
107  token = at(_curr);
108  }
109 
110  TAN_ASSERT(token);
111 
112  ASTBase *node = nullptr;
113  if (token->get_value() == "@") { /// intrinsics
114  node = Intrinsic::Create(_curr);
115  } else if (token->get_value() == "=" && token->get_type() == TokenType::BOP) {
116  node = Assignment::Create(_curr);
117  } else if (token->get_value() == "!") { /// logical not
118  node = UnaryOperator::Create(UnaryOpKind::LNOT, _curr);
119  } else if (token->get_value() == "~") { /// binary not
120  node = UnaryOperator::Create(UnaryOpKind::BNOT, _curr);
121  } else if (token->get_value() == "[") {
122  auto prev = _curr;
123  prev.offset_by(-1);
124  Token *prev_token = at(prev);
125  if (prev_token->get_type() != TokenType::ID && prev_token->get_value() != "]" && prev_token->get_value() != ")") {
126  /// array literal if there is no identifier, "]", or ")" before
127  node = ArrayLiteral::Create(_curr);
128  } else {
129  /// otherwise bracket access
130  node = MemberAccess::Create(_curr);
131  }
132  } else if (token->get_type() == TokenType::RELOP) { /// comparisons
133  BinaryOpKind op = BinaryOpKind::INVALID;
134  switch (hashed_string{token->get_value().c_str()}) {
135  case ">"_hs:
136  op = BinaryOpKind::GT;
137  break;
138  case ">="_hs:
139  op = BinaryOpKind::GE;
140  break;
141  case "<"_hs:
142  op = BinaryOpKind::LT;
143  break;
144  case "<="_hs:
145  op = BinaryOpKind::LE;
146  break;
147  case "=="_hs:
148  op = BinaryOpKind::EQ;
149  break;
150  case "!="_hs:
151  op = BinaryOpKind::NE;
152  break;
153  default:
154  return nullptr;
155  }
156  node = BinaryOperator::Create(op, _curr);
157  } else if (token->get_type() == TokenType::INT) {
158  node = IntegerLiteral::Create(_curr, (uint64_t) std::stol(token->get_value()), token->is_unsigned());
159  } else if (token->get_type() == TokenType::FLOAT) {
160  node = FloatLiteral::Create(_curr, std::stod(token->get_value()));
161  } else if (token->get_type() == TokenType::STRING) { /// string literal
162  node = StringLiteral::Create(_curr, token->get_value());
163  } else if (token->get_type() == TokenType::CHAR) { /// char literal
164  node = CharLiteral::Create(_curr, static_cast<uint8_t>(token->get_value()[0]));
165  } else if (check_typename_token(token)) { /// types, must be before ID
166  node = ASTType::Create(_cs, _curr);
167  } else if (token->get_type() == TokenType::ID) {
168  auto next = _curr;
169  next.offset_by(1);
170  Token *next_token = at(next);
171  if (next_token->get_value() == "(") {
172  /// identifier followed by a "(" is a function call
173  node = FunctionCall::Create(_curr);
174  } else {
175  /// actually an identifier
176  node = Identifier::Create(_curr, token->get_value());
177  }
178  } else if (token->get_type() == TokenType::PUNCTUATION && token->get_value() == "(") {
179  node = Parenthesis::Create(_curr);
180  } else if (token->get_type() == TokenType::KEYWORD) { /// keywords
181  node = peek_keyword(token);
182  if (!node) { report_error(_filename, token, "Keyword not implemented: " + token->to_string()); }
183  } else if (token->get_type() == TokenType::BOP && token->get_value() == ".") { /// member access
184  node = MemberAccess::Create(_curr);
185  } else if (token->get_value() == "&") {
186  /// BOP or UOP? ambiguous
187  node = BinaryOrUnary::Create(_curr, PREC_LOWEST);
188  } else if (token->get_type() == TokenType::PUNCTUATION && token->get_value() == "{") { /// statement(s)
189  node = CompoundStmt::Create(_curr, true);
190  } else if (token->get_type() == TokenType::BOP && check_arithmetic_token(token)) { /// arithmetic operators
191  switch (hashed_string{token->get_value().c_str()}) {
192  case "/"_hs:
193  node = BinaryOperator::Create(BinaryOpKind::DIVIDE, _curr);
194  break;
195  case "%"_hs:
196  node = BinaryOperator::Create(BinaryOpKind::MOD, _curr);
197  break;
198  /// Operators that are possibly BOP or UOP at this stage
199  /// NOTE: using the precedence of the BOP form so that the parsing works correctly if it's really a BOP
200  case "*"_hs:
201  // MULTIPLY / PTR_DEREF
202  node = BinaryOrUnary::Create(_curr, BinaryOperator::BOPPrecedence[BinaryOpKind::MULTIPLY]);
203  break;
204  case "+"_hs:
205  // SUM / PLUS
206  node = BinaryOrUnary::Create(_curr, BinaryOperator::BOPPrecedence[BinaryOpKind::SUM]);
207  break;
208  case "-"_hs:
209  // SUBTRACT / MINUS
210  node = BinaryOrUnary::Create(_curr, BinaryOperator::BOPPrecedence[BinaryOpKind::SUBTRACT]);
211  break;
212  default:
213  TAN_ASSERT(false);
214  return nullptr;
215  }
216  } else if (check_terminal_token(token)) { /// this MUST be the last thing to check
217  return nullptr;
218  } else {
219  report_error(_filename, token, "Unknown token " + token->to_string());
220  }
221  return node;
222  }

224  ASTBase *next_expression(int rbp) {
225  ASTBase *node = peek();
226  if (!node) { return nullptr; }
227  auto n = node;
228  parse_node(n);
229  auto left = n;
230  node = peek();
231  if (!node) { return left; }
232  while (rbp < node->get_bp()) {
233  node = peek();
234  n = node;
235  parse_node(left, n);
236  left = n;
237  node = peek();
238  if (!node) { break; }
239  }
240  return left;
241  }

246  void parse_node(ASTBase *p) {
247  /// special tokens that require whether p is led or nud to determine the node type
248  if (p->get_node_type() == ASTNodeType::BOP_OR_UOP) {
249  auto *pp = ast_must_cast<BinaryOrUnary>(p);
250  UnaryOperator *actual = nullptr;
251  str token_str = _sm->get_token_str(p->get_loc());
252  switch (hashed_string{token_str.c_str()}) {
253  case "*"_hs:
254  actual = UnaryOperator::Create(UnaryOpKind::PTR_DEREF, p->get_loc());
255  break;
256  case "&"_hs:
257  actual = UnaryOperator::Create(UnaryOpKind::ADDRESS_OF, p->get_loc());
258  break;
259  case "+"_hs:
260  actual = UnaryOperator::Create(UnaryOpKind::PLUS, p->get_loc());
261  break;
262  case "-"_hs:
263  actual = UnaryOperator::Create(UnaryOpKind::MINUS, p->get_loc());
264  break;
265  default:
266  TAN_ASSERT(false);
267  break;
268  }
269  pp->set_uop(actual);
270  }
271 
272  switch (p->get_node_type()) {
273  case ASTNodeType::PROGRAM:
274  parse_program(p);
275  break;
276  case ASTNodeType::STATEMENT:
277  parse_stmt(p);
278  break;
279  case ASTNodeType::PARENTHESIS:
280  parse_parenthesis(p);
281  break;
282  case ASTNodeType::IMPORT:
283  parse_import(p);
284  break;
285  case ASTNodeType::INTRINSIC:
286  parse_intrinsic(p);
287  break;
288  case ASTNodeType::IF:
289  parse_if(p);
290  break;
291  case ASTNodeType::LOOP:
292  parse_loop(p);
293  break;
294  case ASTNodeType::UOP:
295  parse_uop(p);
296  break;
297  case ASTNodeType::RET:
298  parse_return(p);
299  break;
300  case ASTNodeType::FUNC_CALL:
301  parse_func_call(p);
302  break;
303  case ASTNodeType::ARRAY_LITERAL:
304  parse_array_literal(p);
305  break;
306  case ASTNodeType::TY:
307  parse_ty(ast_must_cast<ASTType>(p));
308  break;
309  case ASTNodeType::STRUCT_DECL:
310  parse_struct_decl(p);
311  break;
312  case ASTNodeType::VAR_DECL:
313  parse_var_decl(p);
314  break;
315  case ASTNodeType::ARG_DECL:
316  parse_arg_decl(p);
317  break;
318  case ASTNodeType::FUNC_DECL:
319  parse_func_decl(p);
320  break;
321  case ASTNodeType::ENUM_DECL:
322  parse_enum_decl(p);
323  break;
324  case ASTNodeType::BREAK:
325  case ASTNodeType::CONTINUE:
326  case ASTNodeType::ID:
327  case ASTNodeType::INTEGER_LITERAL:
328  case ASTNodeType::FLOAT_LITERAL:
329  case ASTNodeType::CHAR_LITERAL:
330  case ASTNodeType::STRING_LITERAL:
331  case ASTNodeType::BOOL_LITERAL:
332  _curr.offset_by(1);
333  break;
334  case ASTNodeType::BOP_OR_UOP: {
335  auto *pp = ast_must_cast<BinaryOrUnary>(p);
336  TAN_ASSERT(pp->get_kind() == BinaryOrUnary::UNARY);
337 
338  /// update binding power, as the value was originally set to the binding power of BOP version of this op
339  auto *uop = pp->get_uop();
340  uop->set_bp(UnaryOperator::UOPPrecedence[uop->get_op()]);
341 
342  parse_node(pp->get_generic_ptr());
343  break;
344  }
345  default:
346  error(_curr, fmt::format("Unexpected token with type: {}", ASTBase::ASTTypeNames[p->get_node_type()]));
347  break;
348  }

355  /// special tokens that require whether p is led or nud to determine the node type
356  if (p->get_node_type() == ASTNodeType::BOP_OR_UOP) {
357  auto *pp = ast_must_cast<BinaryOrUnary>(p);
358  BinaryOperator *actual = nullptr;
359  str token_str = _sm->get_token_str(p->get_loc());
360  switch (hashed_string{token_str.c_str()}) {
361  case "*"_hs:
362  actual = BinaryOperator::Create(BinaryOpKind::MULTIPLY, p->get_loc());
363  break;
364  case "&"_hs:
365  actual = BinaryOperator::Create(BinaryOpKind::BAND, p->get_loc());
366  break;
367  case "+"_hs:
368  actual = BinaryOperator::Create(BinaryOpKind::SUM, p->get_loc());
369  break;
370  case "-"_hs:
371  actual = BinaryOperator::Create(BinaryOpKind::SUBTRACT, p->get_loc());
372  break;
373  default:
374  TAN_ASSERT(false);
375  break;
376  }
377  pp->set_bop(actual);
378  }
379 
380  switch (p->get_node_type()) {
381  case ASTNodeType::BOP:
382  parse_bop(left, p);
383  break;
384  case ASTNodeType::ASSIGN:
385  parse_assignment(left, p);
386  break;
387  case ASTNodeType::CAST:
388  parse_cast(left, p);
389  break;
390  case ASTNodeType::BOP_OR_UOP: {

393  parse_node(left, pp->get_generic_ptr());
394  break;
395  }
396  default:
397  TAN_ASSERT(false);
398  break;
399  }

536 
537  _curr.offset_by(1); /// skip '['

545  vector<Literal *> elements{};

624  void parse_func_decl(ASTBase *_p) {
625  auto *p = ast_cast<FunctionDecl>(_p);
626 
627  bool is_public = false;
628  bool is_external = false;
629 
630  str token_str = at(_curr)->get_value();
631  if (token_str == "fn") { /// "fn"
632  _curr.offset_by(1);
633  } else if (token_str == "pub") { /// "pub fn"
634  is_public = true;
635  _curr.offset_by(2);
636  } else if (token_str == "extern") { /// "extern"
637  is_external = true;
638  _curr.offset_by(2);
639  } else {
640  TAN_ASSERT(false);
641  }
642 
643  /// function name
644  // Don't use peek since it look ahead and returns ASTNodeType::FUNCTION when it finds "(",
645  // but we only want the function name as an identifier
646  // [X] auto id = peek();
647  Token *id_token = at(_curr);
648  auto id = Identifier::Create(_curr, id_token->get_value());
649  if (id->get_node_type() != ASTNodeType::ID) {
650  error(_curr, "Expect a function name");
651  }
652  parse_node(id);
653  p->set_name(id->get_name());
654 
655  peek(TokenType::PUNCTUATION, "(");
656  _curr.offset_by(1);
657 
658  /// arguments
659  vector<str> arg_names{};
660  vector<ASTType *> arg_types{};
661  vector<ArgDecl *> arg_decls{};
662  if (at(_curr)->get_value() != ")") {
663  while (!eof(_curr)) {
664  auto arg = ArgDecl::Create(_curr);
