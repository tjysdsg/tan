65 ASTBase *peek_keyword(Token *token) {
66 ASTBase *ret = nullptr;
67 str tok = token->get_value();
68 if (tok == "var")
69 ret = VarDecl::Create(_sm->src());
70 else if (tok == "fn" || tok == "pub" || tok == "extern")
71 ret = FunctionDecl::Create(_sm->src());
72 else if (tok == "import")
73 ret = Import::Create(_sm->src());
74 else if (tok == "if") /// else clause should be covered by If statement as well
75 ret = If::Create(_sm->src());
76 else if (tok == "return")
77 ret = Return::Create(_sm->src());
78 else if (tok == "while" || tok == "for")
79 ret = Loop::Create(_sm->src());
80 else if (tok == "struct")
81 ret = StructDecl::Create(_sm->src());
82 else if (tok == "break")
83 ret = Break::Create(_sm->src());
84 else if (tok == "continue")
85 ret = Continue::Create(_sm->src());
86 else if (tok == "as")
87 ret = Cast::Create(_sm->src());
88 else if (tok == "true")
89 ret = BoolLiteral::Create(_sm->src(), true);
90 else if (tok == "false")
91 ret = BoolLiteral::Create(_sm->src(), false);
92 else if (tok == "package")

103 ASTBase *peek() {
104 if (eof(_curr)) {
105 return nullptr;
106 }
107 Token *token = at(_curr);
108 /// skip comments
109 while (token && token->get_type() == TokenType::COMMENTS) {
110 ++_curr;
111 if (eof(_curr)) {
112 return nullptr;
113 }
114 token = at(_curr);
115 }
116
117 TAN_ASSERT(token);
118
119 ASTBase *node = nullptr;
120 if (token->get_value() == "@") { /// intrinsics
121 node = Intrinsic::Create(_sm->src());
122 } else if (token->get_value() == "=" && token->get_type() == TokenType::BOP) {
123 node = Assignment::Create(_sm->src());
124 } else if (token->get_value() == "!") { /// logical not
125 node = UnaryOperator::Create(UnaryOpKind::LNOT, _sm->src());
126 } else if (token->get_value() == "~") { /// binary not
127 node = UnaryOperator::Create(UnaryOpKind::BNOT, _sm->src());
128 } else if (token->get_value() == "[") {
129 auto prev = _curr;
130 Token *prev_token = at(--prev);
131 if (prev_token->get_type() != TokenType::ID && prev_token->get_value() != "]" && prev_token->get_value() != ")") {
132 /// array literal if there is no identifier, "]", or ")" before
133 node = ArrayLiteral::Create(_sm->src());
134 } else {
135 /// otherwise bracket access
136 node = MemberAccess::Create(_sm->src());
137 }
138 } else if (token->get_type() == TokenType::RELOP) { /// comparisons
139 BinaryOpKind op;
140 str tok = token->get_value();
141 if (tok == ">")
142 op = BinaryOpKind::GT;
143 else if (tok == ">=")
144 op = BinaryOpKind::GE;
145 else if (tok == "<")
146 op = BinaryOpKind::LT;
147 else if (tok == "<=")
148 op = BinaryOpKind::LE;
149 else if (tok == "==")
150 op = BinaryOpKind::EQ;
151 else if (tok == "!=")
152 op = BinaryOpKind::NE;
153 else if (tok == "&&")
154 op = BinaryOpKind::LAND;
155 else if (tok == "||")
156 op = BinaryOpKind::LOR;
157 else
158 error(ErrorType::NOT_IMPLEMENTED, _curr, _curr,
159 fmt::format("Binary relational operator not implemented: {}", token->get_value().c_str()));
160
161 node = BinaryOperator::Create(op, _sm->src());
162 } else if (token->get_type() == TokenType::INT) {
163 node = IntegerLiteral::Create(_sm->src(), (uint64_t)std::stol(token->get_value()), token->is_unsigned());
164 } else if (token->get_type() == TokenType::FLOAT) {
165 node = FloatLiteral::Create(_sm->src(), std::stod(token->get_value()));
166 } else if (token->get_type() == TokenType::STRING) { /// string literal
167 node = StringLiteral::Create(_sm->src(), token->get_value());
168 } else if (token->get_type() == TokenType::CHAR) { /// char literal
169 node = CharLiteral::Create(_sm->src(), static_cast<uint8_t>(token->get_value()[0]));
170 } else if (check_typename_token(token)) { /// should not encounter types if parsed properly
171 TAN_ASSERT(false);
172 } else if (token->get_type() == TokenType::ID) {
173 auto next = _curr;
174 Token *next_token = at(++next);
175 if (next_token->get_value() == "(") {
176 /// identifier followed by a "(" is a function call
177 node = FunctionCall::Create(_sm->src());
178 } else {
179 /// actually an identifier
180 node = Identifier::Create(_sm->src(), token->get_value());
181 }
182 } else if (token->get_type() == TokenType::PUNCTUATION && token->get_value() == "(") {
183 node = Parenthesis::Create(_sm->src());
184 } else if (token->get_type() == TokenType::KEYWORD) { /// keywords
185 node = peek_keyword(token);
186 if (!node) {
187 error(ErrorType::NOT_IMPLEMENTED, _curr, _curr, "Keyword not implemented: " + token->get_value());
188 }
189 } else if (token->get_type() == TokenType::BOP && token->get_value() == ".") { /// member access
190 node = MemberAccess::Create(_sm->src());
191 } else if (token->get_value() == "&") {
192 /// BOP or UOP? ambiguous
193 node = BinaryOrUnary::Create(_sm->src(), BinaryOperator::BOPPrecedence[BinaryOpKind::BAND]);
194 } else if (token->get_type() == TokenType::PUNCTUATION && token->get_value() == "{") { /// statement(s)
195 node = CompoundStmt::Create(_sm->src());
196 } else if (token->get_type() == TokenType::BOP) { /// binary operators that haven't been processed yet
197 TAN_ASSERT(token->get_value().length());
198 switch (token->get_value()[0]) {
199 case '/':
200 node = BinaryOperator::Create(BinaryOpKind::DIVIDE, _sm->src());
201 break;
202 case '%':
203 node = BinaryOperator::Create(BinaryOpKind::MOD, _sm->src());
204 break;
205 case '|':
206 node = BinaryOperator::Create(BinaryOpKind::BOR, _sm->src());
207 break;
208 case '^':
209 node = BinaryOperator::Create(BinaryOpKind::XOR, _sm->src());
210 break;
211 /// Operators that are possibly BOP or UOP at this stage
212 /// NOTE: using the precedence of the BOP form so that the parsing works correctly if it's really a BOP
213 case '*':
214 // MULTIPLY / PTR_DEREF
215 node = BinaryOrUnary::Create(_sm->src(), BinaryOperator::BOPPrecedence[BinaryOpKind::MULTIPLY]);
216 break;
217 case '+':
218 // SUM / PLUS
219 node = BinaryOrUnary::Create(_sm->src(), BinaryOperator::BOPPrecedence[BinaryOpKind::SUM]);
220 break;
221 case '-':
222 // SUBTRACT / MINUS
223 node = BinaryOrUnary::Create(_sm->src(), BinaryOperator::BOPPrecedence[BinaryOpKind::SUBTRACT]);
224 break;
225 default:
226 TAN_ASSERT(false);
227 return nullptr;
228 }
229 } else if (check_terminal_token(token)) { /// this MUST be the last thing to check
230 return nullptr;
231 } else {
232 error(ErrorType::SYNTAX_ERROR, _curr, _curr, "Unknown token " + token->get_value());
233 }
234
235 node->set_start(_curr);
236 node->set_end(_curr);
237 return node;
238 }

240 ASTBase *next_expression(int rbp) {
241 ASTBase *node = peek();
242 if (!node) {
243 return nullptr;
244 }
245 auto n = node;
246 parse_node(n);
247 auto left = n;
248 node = peek();
249 if (!node) {
250 return left;
251 }
252 while (rbp < node->get_bp()) {
253 node = peek();
254 n = node;
255 parse_node(left, n);
256 left = n;
257 node = peek();
258 if (!node) {
259 break;
260 }
261 }
262 return left;
263 }

265 void parse_program(Program *p) {
266 ScopeGuard scope_guard(_curr_scope, p);
267
268 while (!eof(_curr)) {
269 ASTBase *node = next_expression(PREC_LOWEST);
270 if (!node)
271 error(ErrorType::SYNTAX_ERROR, _curr, _curr, "Unexpected terminal token");
272
273 p->append_child(node);
274 if (!check_terminal_token(at(_curr))) {
275 error(ErrorType::SYNTAX_ERROR, _curr, _curr, "Expect a terminal token");
276 }
277
278 ++_curr;
279 }

282 /// Parse NUD
283 void parse_node(ASTBase *p) {
284 /// special tokens that require whether p is led or nud to determine the node type
285 if (p->get_node_type() == ASTNodeType::BOP_OR_UOP) {
286 auto *pp = ast_cast<BinaryOrUnary>(p);
287 UnaryOperator *actual = nullptr;
288 str tok = _sm->get_token_str(p->start());
289 TAN_ASSERT(tok.length());
290 switch (tok[0]) {
291 case '*':
292 actual = UnaryOperator::Create(UnaryOpKind::PTR_DEREF, _sm->src());
293 break;
294 case '&':
295 actual = UnaryOperator::Create(UnaryOpKind::ADDRESS_OF, _sm->src());
296 break;
297 case '+':
298 actual = UnaryOperator::Create(UnaryOpKind::PLUS, _sm->src());
299 break;
300 case '-':
301 actual = UnaryOperator::Create(UnaryOpKind::MINUS, _sm->src());
302 break;
303 default:
304 TAN_ASSERT(false);
305 break;
306 }
307 pp->set_uop(actual);
308
309 parse_node(pp->get_expr_ptr());
310 return;
311 }
312
313 // look up parser func from the table
314 auto it = NUD_PARSING_FUNC_TABLE.find(p->get_node_type());
315 if (it == NUD_PARSING_FUNC_TABLE.end()) {
316 error(ErrorType::SYNTAX_ERROR, _curr, _curr,
317 fmt::format("Unexpected token with type: {}", ASTBase::ASTTypeNames[p->get_node_type()]));
318 }
319 nud_parsing_func_t func = it->second;
320 (this->*func)(p);
321 }

323 /// Parse LED
324 void parse_node(ASTBase *left, ASTBase *p) {
325 /// special tokens that require whether p is led or nud to determine the node type
326 if (p->get_node_type() == ASTNodeType::BOP_OR_UOP) {
327 auto *pp = ast_cast<BinaryOrUnary>(p);
328 BinaryOperator *actual = nullptr;
329 str tok = _sm->get_token_str(p->start());
330 TAN_ASSERT(tok.length());
331 switch (tok[0]) {
332 case '*':
333 actual = BinaryOperator::Create(BinaryOpKind::MULTIPLY, _sm->src());
334 break;
335 case '&':
336 actual = BinaryOperator::Create(BinaryOpKind::BAND, _sm->src());
337 break;
338 case '+':
339 actual = BinaryOperator::Create(BinaryOpKind::SUM, _sm->src());
340 break;
341 case '-':
342 actual = BinaryOperator::Create(BinaryOpKind::SUBTRACT, _sm->src());
343 break;
344 default:
345 TAN_ASSERT(false);
346 break;
347 }
348 pp->set_bop(actual);
349 parse_node(left, pp->get_expr_ptr());
350 return;
351 }
352
353 // look up parser func from the table
354 auto it = LED_PARSING_FUNC_TABLE.find(p->get_node_type());
355 if (it == LED_PARSING_FUNC_TABLE.end()) {
356 error(ErrorType::SYNTAX_ERROR, _curr, _curr,
357 fmt::format("Unexpected token with type: {}", ASTBase::ASTTypeNames[p->get_node_type()]));
358 }
359 led_parsing_func_t func = it->second;
360 (this->*func)(left, p);
361 }

430 void parse_if(ASTBase *_p) {
431 auto p = ast_cast<If>(_p);
432
433 p->set_end(_curr); // the end of token of If AST ends here
434
435 /// if then
436 parse_if_then_branch(p);
437 ++_curr; // skip "}"
438
439 /// else or elif clause, if any
440 while (at(_curr)->get_value() == "else") {
441 ++_curr; // skip "else"
442 if (at(_curr)->get_value() == "if") { /// elif
443 parse_if_then_branch(p);
444 } else if (at(_curr)->get_value() == "{") { /// else
445 auto else_clause = peek();
446 parse_node(else_clause);
447 p->add_else_branch(expect_stmt(else_clause));
448 } else {
449 error(ErrorType::SYNTAX_ERROR, _curr, _curr, "Unexpected token");
450 }
451
452 ++_curr;
453 }
454
455 --_curr; /// return back to "}"
456 TAN_ASSERT(at(_curr)->get_value() == "}");
457 }

459 void parse_if_then_branch(If *p) {
460 ++_curr; // skip "if"
461
462 /// predicate
463 auto _pred = peek("(");
464 if (_pred->get_node_type() != ASTNodeType::PARENTHESIS) {
465 error(ErrorType::SYNTAX_ERROR, _curr, _curr, "Expect a parenthesis expression");
466 }
467 parse_parenthesis(_pred);
468 Expr *pred = ast_cast<Expr>(_pred);
469
470 /// then clause
471 auto *_then = peek("{");
472 parse_node(_then);
473 Stmt *then_clause = expect_stmt(_then);
474
475 p->add_if_then_branch(pred, then_clause);
476 }

478 void parse_loop(ASTBase *_p) {
479 auto p = ast_cast<Loop>(_p);
480
481 if (at(_curr)->get_value() == "for") {

514 void parse_array_literal(ASTBase *_p) {
515 auto *p = ast_cast<ArrayLiteral>(_p);
516
517 ++_curr; // skip '['
518
519 if (at(_curr)->get_value() == "]") {
520 // TODO: support empty array literal, but raise error if the type cannot be inferred
521 error(ErrorType::SEMANTIC_ERROR, p->start(), _curr, "Empty array literal");
522 }
523
524 vector<Literal *> elements{};
525 while (!eof(_curr)) {
526 if (at(_curr)->get_value() == ",") { /// skip ","
527 ++_curr;
528 continue;
529 } else if (at(_curr)->get_value() == "]") { /// skip "]"
530 ++_curr;
531 break;
532 }
533
534 auto *node = peek();
535 auto *expr = expect_expression(node);
536 if (!expr->is_comptime_known()) {
537 error(ErrorType::SEMANTIC_ERROR, _curr, _curr, "Expected a compile-time known value");
538 }
539
540 parse_node(node);
541 elements.push_back(ast_cast<Literal>(node));
542 }
543
544 p->set_elements(elements);
545 p->set_end(_curr - 1);
546 }

581 void parse_parenthesis(ASTBase *_p) {

583
584 ++_curr; // skip "("
585 while (true) {
586 auto *t = at(_curr);
587 if (t->get_type() == TokenType::PUNCTUATION && t->get_value() == ")") { /// end at )
588 ++_curr;

595 Expr *sub = expect_expression(_sub);
596 p->set_sub(sub);
597 }

602 void parse_func_decl(ASTBase *_p) {
603 auto *p = ast_cast<FunctionDecl>(_p);
604
605 bool is_public = false;
606 bool is_external = false;
607 str token_str = at(_curr)->get_value();
608 if (token_str == "fn") { /// "fn"
609 ++_curr;
610 } else if (token_str == "pub") { /// "pub fn"
611 is_public = true;
612 _curr += 2;
613 } else if (token_str == "extern") { /// "extern"
614 is_external = true;
615 _curr += 2;
616 } else {
617 TAN_ASSERT(false);
618 }
619 p->set_public(is_public);
620 p->set_external(is_external);
621
622 /// function name
623 // Don't use peek since it look ahead and returns ASTNodeType::FUNCTION when it finds "(",
624 // but we only want the function name as an identifier
625 // [X] auto id = peek();
626 Token *id_token = at(_curr);
627 auto id = Identifier::Create(_sm->src(), id_token->get_value());
628 id->set_start(_curr);
629 id->set_end(_curr);
630 if (id->get_node_type() != ASTNodeType::ID) {
631 error(ErrorType::SYNTAX_ERROR, _curr, _curr, "Expect a function name");
632 }
633 parse_node(id);
634 p->set_name(id->get_name());
635
636 peek("(");
637 ++_curr;
638
639 // Register in the parent context
640 if (is_public) {
641 p->ctx()->set_function_decl(p);
642 } else {
643 p->ctx()->set_function_decl(p);
644 }
645
646 { // declarations of func arguments and variables in the body are within the local scope
647 ScopeGuard scope_guard(_curr_scope, p);
648
649 /// arguments
650 vector<str> arg_names{};
651 vector<Type *> arg_types{};
652 vector<ArgDecl *> arg_decls{};
653 if (at(_curr)->get_value() != ")") {
654 while (!eof(_curr)) {
655 auto arg = ArgDecl::Create(_sm->src());
656 arg->set_start(_curr);
657 parse_node(arg);
658
659 arg_names.push_back(arg->get_name());
660 arg_types.push_back(arg->get_type());
661 arg_decls.push_back(arg);
662
663 if (at(_curr)->get_value() == ",") {
664 ++_curr;
665 } else {
666 break;
667 }
668 }
669 }
670
671 peek(")");
672 p->set_end(_curr);
673 ++_curr;
674
675 p->set_arg_names(arg_names);
676 p->set_arg_decls(arg_decls);
677
678 peek(":");
679 ++_curr;
680
681 /// function type
682 auto *ret_type = peek_type();
683 auto *func_type = Type::GetFunctionType(parse_ty(ret_type), arg_types);
684 p->set_type(func_type);
685
686 /// body
687 if (!is_external) {
688 auto body = peek("{");
689 parse_node(body);
690 p->set_body(expect_stmt(body));
691 }
692 }
693 }

695 void parse_func_call(ASTBase *_p) {
696 auto *p = ast_cast<FunctionCall>(_p);
697
698 p->set_name(at(_curr)->get_value()); // function name
699 ++_curr;
700
701 // No need to check since '(' is what distinguish a function call from an identifier at the first place
702 // auto *token = at(_curr); if (token->get_value() != "(") { error("Invalid function call"); }
703 ++_curr; // skip (
704
705 // args
706 while (!eof(_curr) && at(_curr)->get_value() != ")") {
707 auto _arg = next_expression(PREC_LOWEST);
708 Expr *arg = expect_expression(_arg);
709 p->_args.push_back(arg);
710
711 if (at(_curr)->get_value() == ",") { /// skip ,
712 ++_curr;
713 } else {
714 break;
715 }
716 }
717
718 peek(")");
719 p->set_end(_curr);
720 ++_curr;

757 void parse_intrinsic(ASTBase *_p) {
758 auto *p = ast_cast<Intrinsic>(_p);
759
760 ++_curr; // skip "@"
761
762 if (_sm->get_token_str(_curr) == "test_comp_error") {
763 parse_test_comp_error_intrinsic(p);
764 return;
765 }
766
767 auto e = peek();
768 parse_node(e);
769 // Only allow identifier or function call as the valid intrinsic token
770 if (e->get_node_type() != ASTNodeType::ID && e->get_node_type() != ASTNodeType::FUNC_CALL) {
771 error(ErrorType::SYNTAX_ERROR, _curr, _curr, "Unexpected token");
772 }
773 p->set_sub(e);
774
775 // find out the name
776 str name;
777 switch (e->get_node_type()) {
778 case ASTNodeType::FUNC_CALL:
779 name = ast_cast<FunctionCall>(e)->get_name();
780 break;
781 case ASTNodeType::ID:
782 name = ast_cast<Identifier>(e)->get_name();
783 break;
784 default:
785 TAN_ASSERT(false);
786 break;
787 }
788 TAN_ASSERT(!name.empty());
789
790 // figure out which intrinsic
791 p->set_name(name);
792 auto q = Intrinsic::intrinsics.find(name);
793 if (q == Intrinsic::intrinsics.end()) {
794 error(ErrorType::UNKNOWN_SYMBOL, _curr, _curr, fmt::format("Unknown intrinsic {}", name));
795 }
796 p->set_intrinsic_type(q->second);
797
798 // check if the syntax is correct
799 switch (p->get_intrinsic_type()) {
800 case IntrinsicType::LINENO:
801 case IntrinsicType::FILENAME:
802 if (e->get_node_type() != ASTNodeType::ID)
803 error(ErrorType::SEMANTIC_ERROR, _curr, _curr, "Invalid usage of intrinsic");
804 break;
805 case IntrinsicType::NOOP:
806 case IntrinsicType::STACK_TRACE:
807 case IntrinsicType::TEST_COMP_ERROR:
808 if (e->get_node_type() != ASTNodeType::FUNC_CALL)
809 error(ErrorType::SEMANTIC_ERROR, _curr, _curr, "Invalid usage of intrinsic");
810 break;
811 case IntrinsicType::INVALID:
812 TAN_ASSERT(false);
813 break;
814 default:
815 break;
816 }
817
818 p->set_end(_curr - 1);
819 }

913 void parse_struct_decl(ASTBase *_p) {
914 auto *p = ast_cast<StructDecl>(_p);
915
916 ++_curr; // skip "struct"
917
918 // struct typename
919 auto _id = peek();
920 if (_id->get_node_type() != ASTNodeType::ID) {
921 error(ErrorType::SYNTAX_ERROR, _curr, _curr, "Expecting a typename");
922 }
923 parse_node(_id);
924 auto id = ast_cast<Identifier>(_id);
925 p->set_name(id->get_name());
926
927 p->set_end(_curr - 1);
928
929 // struct body
930 if (at(_curr)->get_value() != "{") {
931 error(ErrorType::SYNTAX_ERROR, _curr, _curr, "Expect struct body");
932 }
933
934 ScopeGuard scope_guard(_curr_scope, p);
935
936 auto _comp_stmt = next_expression(PREC_LOWEST);
937 if (!_comp_stmt || _comp_stmt->get_node_type() != ASTNodeType::COMPOUND_STATEMENT) {
938 error(ErrorType::SEMANTIC_ERROR, _curr, _curr, "struct definition requires a valid body");
939 }
940 auto comp_stmt = ast_cast<CompoundStmt>(_comp_stmt);
941
942 // copy member declarations
943 auto children = comp_stmt->get_children();
944 vector<Expr *> member_decls{};
945 for (const auto &c : children) {
946 if (!( //
947 c->get_node_type() == ASTNodeType::VAR_DECL //
948 || c->get_node_type() == ASTNodeType::ASSIGN //
949 || c->get_node_type() == ASTNodeType::FUNC_DECL //
950 )) {
951 error(ErrorType::SEMANTIC_ERROR, c, "Invalid struct member");
952 }
953 member_decls.push_back(ast_cast<Expr>(c));
954 }
955 p->set_member_decls(member_decls);
956 }

958 ArrayType *parse_ty_array(Type *p) {
959 ArrayType *ret = nullptr;
960 while (true) {
961 ++_curr; // skip "["
962
963 // size
964 ASTBase *_size = peek();
965 if (_size->get_node_type() != ASTNodeType::INTEGER_LITERAL) {
966 error(ErrorType::SYNTAX_ERROR, _curr, _curr, "Expect an unsigned integer as the array size");
967 }
968 parse_node(_size);
969
970 auto size = ast_cast<IntegerLiteral>(_size);
971 size_t array_size = size->get_value();
972 if (static_cast<int64_t>(array_size) < 0) {
973 error(ErrorType::SYNTAX_ERROR, _curr, _curr, "Expect an unsigned integer as the array size");
974 }
975
976 ret = Type::GetArrayType(p, (int)array_size);
977
978 // skip "]"
979 peek("]");
980 ++_curr;
981
982 // if followed by a "[", this is a multi-dimension array
983 if (at(_curr)->get_value() == "[") {
984 p = ret;
985 ret = nullptr;
986 } else {
987 break;
988 }
989 }
990
991 TAN_ASSERT(ret);
992 return ret;
993 }

995 Type *parse_ty(Type *p) {
996 Type *ret = p;
997
998 while (!eof(_curr)) {
999 Token *token = at(_curr);
1000
1001 auto it = PrimitiveType::TYPENAME_TO_KIND.find(token->get_value());
1002 if (it != PrimitiveType::TYPENAME_TO_KIND.end()) { // primitive
1003 ret = PrimitiveType::Create(it->second);
1004 } else if (token->get_value() == "*") { // pointer
1005 TAN_ASSERT(ret);
1006 ret = Type::GetPointerType(ret);
1007 } else if (token->get_value() == "str") {
1008 ret = Type::GetStringType();
1009 } else if (token->get_type() == TokenType::ID) { // struct/typedefs etc.
1010 /// type referred will be resolved in analysis phase
1011 } else {
1012 break;
1013 }
1014 ++_curr;
1015 }
1016
1017 // array
1018 Token *token = at(_curr);
1019 if (token->get_value() == "[") {
1020 TAN_ASSERT(ret);
1021 ret = parse_ty_array(ret);
1022 }
1023
1024 return ret;
1025 }
