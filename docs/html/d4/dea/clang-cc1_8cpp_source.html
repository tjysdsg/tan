23 static void LLVMErrorHandler(void *UserData, const std::string &Message, bool GenCrashDiag) {
24  DiagnosticsEngine &Diags = *static_cast<DiagnosticsEngine *>(UserData);
25 
26  Diags.Report(diag::err_fe_error_backend) << Message;
27 
28  // Run the interrupt handlers to make sure any special cleanups get done, in
29  // particular that we remove files registered with RemoveFileOnSignal.
30  llvm::sys::RunInterruptHandlers();
31 
32  // We cannot recover from llvm errors. When reporting a fatal error, exit
33  // with status 70 to generate crash diagnostics. For BSD systems this is
34  // defined as an internal software error. Otherwise, exit with status 1.
35  llvm::sys::Process::Exit(GenCrashDiag ? 70 : 1);
36 }

41 static size_t getCurrentStackAllocation() {
42  // If we can't compute the current stack usage, allow for 512K of command
43  // line arguments and environment.
44  size_t Usage = 512 * 1024;
45  if (FILE *StatFile = fopen("/proc/self/stat", "r")) {
46  // We assume that the stack extends from its current address to the end of
47  // the environment space. In reality, there is another string literal (the
48  // program name) after the environment, but this is close enough (we only
49  // need to be within 100K or so).
50  unsigned long StackPtr, EnvEnd;
51  // Disable silly GCC -Wformat warning that complains about length
52  // modifiers on ignored format specifiers. We want to retain these
53  // for documentation purposes even though they have no effect.
54  #if defined(__GNUC__) && !defined(__clang__)
55  #pragma GCC diagnostic push
56  #pragma GCC diagnostic ignored "-Wformat"
57  #endif
58  if (fscanf(StatFile, "%*d %*s %*c %*d %*d %*d %*d %*d %*u %*lu %*lu %*lu %*lu %*lu "
59  "%*lu %*ld %*ld %*ld %*ld %*ld %*ld %*llu %*lu %*ld %*lu %*lu "
60  "%*lu %*lu %lu %*lu %*lu %*lu %*lu %*lu %*llu %*lu %*lu %*d %*d "
61  "%*u %*u %*llu %*lu %*ld %*lu %*lu %*lu %*lu %*lu %*lu %lu %*d", &StackPtr, &EnvEnd) == 2) {
62  #if defined(__GNUC__) && !defined(__clang__)
63  #pragma GCC diagnostic pop
64  #endif
65  Usage = StackPtr < EnvEnd ? EnvEnd - StackPtr : StackPtr - EnvEnd;
66  }
67  fclose(StatFile);
68  }
69  return Usage;
70 }

75 static void ensureStackAddressSpace() {
76  // Linux kernels prior to 4.1 will sometimes locate the heap of a PIE binary
77  // relatively close to the stack (they are only guaranteed to be 128MiB
78  // apart). This results in crashes if we happen to heap-allocate more than
79  // 128MiB before we reach our stack high-water mark.
80  //
81  // To avoid these crashes, ensure that we have sufficient virtual memory
82  // pages allocated before we start running.
83  size_t Curr = getCurrentStackAllocation();
84  const int kTargetStack = DesiredStackSize - 256 * 1024;
85  if (Curr < kTargetStack) {
86  volatile char *volatile Alloc = static_cast<volatile char *>(alloca(kTargetStack - Curr));
87  Alloc[0] = 0;
88  Alloc[kTargetStack - Curr - 1] = 0;
89  }
90 }

96 /// Attempt to ensure that we have at least 8MiB of usable stack space.
97 static void ensureSufficientStack() {
98  struct rlimit rlim;
99  if (getrlimit(RLIMIT_STACK, &rlim) != 0) {
100  return;
101  }
102 
103  // Increase the soft stack limit to our desired level, if necessary and
104  // possible.
105  if (rlim.rlim_cur != RLIM_INFINITY && rlim.rlim_cur < rlim_t(DesiredStackSize)) {
106  // Try to allocate sufficient stack.
107  if (rlim.rlim_max == RLIM_INFINITY || rlim.rlim_max >= rlim_t(DesiredStackSize)) {
108  rlim.rlim_cur = DesiredStackSize;
109  } else if (rlim.rlim_cur == rlim.rlim_max) {
110  return;
111  } else {
112  rlim.rlim_cur = rlim.rlim_max;
113  }
114 
115  if (setrlimit(RLIMIT_STACK, &rlim) != 0 || rlim.rlim_cur != DesiredStackSize) {
116  return;
117  }
118  }
119 
120  // We should now have a stack of size at least DesiredStackSize. Ensure
121  // that we can actually use that much, if necessary.
122  ensureStackAddressSpace();
123 }

129 /// Print supported cpus of the given target.
130 static int PrintSupportedCPUs(std::string TargetStr) {
131  std::string Error;
132  const llvm::Target *TheTarget = llvm::TargetRegistry::lookupTarget(TargetStr, Error);
133  if (!TheTarget) {
134  llvm::errs() << Error;
135  return 1;
136  }
137 
138  // the target machine will handle the mcpu printing
139  llvm::TargetOptions Options;
140  std::unique_ptr<llvm::TargetMachine>
141  TheTargetMachine(TheTarget->createTargetMachine(TargetStr, "", "+cpuHelp", Options, None));
142  return 0;
143 }

145 int cc1_main(ArrayRef<const char *> Argv, const char *Argv0, void *MainAddr) {
146  ensureSufficientStack();
147 
148  std::unique_ptr<CompilerInstance> Clang(new CompilerInstance());
149  IntrusiveRefCntPtr<DiagnosticIDs> DiagID(new DiagnosticIDs());
150 
151  // Register the support for object-file-wrapped Clang modules.
152  auto PCHOps = Clang->getPCHContainerOperations();
153  PCHOps->registerWriter(std::make_unique<ObjectFilePCHContainerWriter>());
154  PCHOps->registerReader(std::make_unique<ObjectFilePCHContainerReader>());
155 
156  // Initialize targets first, so that --version shows registered targets.
157  llvm::InitializeAllTargets();
158  llvm::InitializeAllTargetMCs();
159  llvm::InitializeAllAsmPrinters();
160  llvm::InitializeAllAsmParsers();
161 
162  // Buffer diagnostics from argument parsing so that we can output them using a
163  // well formed diagnostic object.
164  IntrusiveRefCntPtr<DiagnosticOptions> DiagOpts = new DiagnosticOptions();
165  TextDiagnosticBuffer *DiagsBuffer = new TextDiagnosticBuffer;
166  DiagnosticsEngine Diags(DiagID, &*DiagOpts, DiagsBuffer);
167  bool Success = CompilerInvocation::CreateFromArgs(Clang->getInvocation(), Argv, Diags);
168 
169  if (Clang->getFrontendOpts().TimeTrace) {
170  llvm::timeTraceProfilerInitialize(Clang->getFrontendOpts().TimeTraceGranularity, Argv0);
171  }
172  // --print-supported-cpus takes priority over the actual compilation.
173  if (Clang->getFrontendOpts().PrintSupportedCPUs) {
174  return PrintSupportedCPUs(Clang->getTargetOpts().Triple);
175  }
176 
177  // Infer the builtin include path if unspecified.
178  if (Clang->getHeaderSearchOpts().UseBuiltinIncludes && Clang->getHeaderSearchOpts().ResourceDir.empty()) {
179  Clang->getHeaderSearchOpts().ResourceDir = CompilerInvocation::GetResourcesPath(Argv0, MainAddr);
180  }
181 
182  // Create the actual diagnostics engine.
183  Clang->createDiagnostics();
184  if (!Clang->hasDiagnostics()) {
185  return 1;
186  }
187 
188  // Set an error handler, so that any LLVM backend diagnostics go through our
189  // error handler.
190  llvm::install_fatal_error_handler(LLVMErrorHandler, static_cast<void *>(&Clang->getDiagnostics()));
191 
192  DiagsBuffer->FlushDiagnostics(Clang->getDiagnostics());
193  if (!Success) {
194  return 1;
195  }
196 
197  // Execute the frontend actions.
198  {
199  llvm::TimeTraceScope TimeScope("ExecuteCompiler");
200  Success = ExecuteCompilerInvocation(Clang.get());
201  }
202 
203  // If any timers were active but haven't been destroyed yet, print their
204  // results now. This happens in -disable-free mode.
205  llvm::TimerGroup::printAll(llvm::errs());
206  llvm::TimerGroup::clearAll();
207 
208  if (llvm::timeTraceProfilerEnabled()) {
209  SmallString<128> Path(Clang->getFrontendOpts().OutputFile);
210  llvm::sys::path::replace_extension(Path, "json");
211  if (auto profilerOutput = Clang->createOutputFile(Path.str(),
212  /*Binary=*/false,
213  /*RemoveFileOnSignal=*/false, "",
214  /*Extension=*/"json",
215  /*useTemporary=*/false)) {
216 
217  llvm::timeTraceProfilerWrite(*profilerOutput);
218  // FIXME(ibiryukov): make profilerOutput flush in destructor instead.
219  profilerOutput->flush();
220  llvm::timeTraceProfilerCleanup();
221  }
222  }
223 
224  // Our error handler depends on the Diagnostics object, which we're
225  // potentially about to delete. Uninstall the handler now so that any
226  // later errors use the default handling behavior instead.
227  llvm::remove_fatal_error_handler();
228 
229  // When running with -disable-free, don't do any destruction or shutdown.
230  if (Clang->getFrontendOpts().DisableFree) {
231  llvm::BuryPointer(std::move(Clang));
232  return !Success;
233  }
234 
235  return !Success;
236 }
