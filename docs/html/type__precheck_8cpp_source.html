20void TypePrecheck::run_impl(Package *p) {
21 _package = p;
22
23 push_scope(p);
24
25 for (const auto &c : p->get_children()) {
26 switch (c->get_node_type()) {
27 case ASTNodeType::IMPORT:
28 CALL_AST_VISITOR(Import, c);
29 break;
30 case ASTNodeType::STRUCT_DECL:
31 CALL_AST_VISITOR(StructDecl, c);
32 break;
33 case ASTNodeType::FUNC_DECL:
34 CALL_AST_VISITOR(FunctionDecl, c);
35 break;
36 case ASTNodeType::INTRINSIC:
37 CALL_AST_VISITOR(Intrinsic, c);
38 break;
39 default:
40 break;
41 }
42 }
43
44 pop_scope();
45}

47Type *TypePrecheck::check_type_ref(Type *p, ASTBase *node) {
48 TAN_ASSERT(p->is_ref());
49 Type *ret = p;
50
51 const str &referred_name = p->get_typename();
52 auto *decl = search_decl_in_scopes(referred_name);
53 if (decl && decl->is_type_decl()) {
54 ret = decl->get_type();
55 TAN_ASSERT(ret);
56 if (!ret->is_canonical()) {
57 _package->top_level_symbol_dependency.add_dependency(decl, node);
58 }
59 } else {
60 error(ErrorType::TYPE_ERROR, node, fmt::format("Unknown type {}", referred_name));
61 }
62
63 return ret;
64}

66Type *TypePrecheck::check_type(Type *p, ASTBase *node) {
67 TAN_ASSERT(p);
68 TAN_ASSERT(node);
69
70 Type *ret = p;
71 if (p->is_ref()) {
72 ret = check_type_ref(p, node);
73 } else if (p->is_pointer()) {
74 auto *pointee = pcast<PointerType>(p)->get_pointee();
75
76 TAN_ASSERT(pointee);
77 if (pointee->is_ref()) {
78 pointee = check_type_ref(pointee, node);
79 if (pointee->is_canonical()) {
80 ret = Type::GetPointerType(pointee);
81 }
82 }
83 }
84
85 TAN_ASSERT(ret);
86 return ret;
87}

89DEFINE_AST_VISITOR_IMPL(TypePrecheck, Import) {
90 str name = p->get_name();
91
92 auto *compiler = CompilerDriver::instance();
93 TAN_ASSERT(compiler);
94 Package *package = compiler->get_package(name);
95 if (!package) {
96 error(ErrorType::IMPORT_ERROR, p, "Cannot find package named: " + name);
97 }
98
99 Context *imported_ctx = package->ctx();
100
101 // Imported declarations are stored in both the package's context and this import AST node:
102 // 1. The context only serves as a search table
103 // 2. The type checker and code generator will operate on the declarations stored in the import node
104
105 // import functions
106 vector<FunctionDecl *> funcs = imported_ctx->get_func_decls();
107 vector<FunctionDecl *> pub_funcs{};
108 for (auto *f : funcs) {
109 f->set_start(p->start());
110 f->set_end(p->end());
111
112 if (f->is_public() || f->is_external()) {
113 auto *existing = top_ctx()->get_func_decl(f->get_name());
114 if (!existing) {
115 // FIXME: merge multiple declarations of the same symbol, fail if they don't match
116 pub_funcs.push_back(f);
117 top_ctx()->set_function_decl(f);
118 }
119 }
120 }
121 p->_imported_funcs = pub_funcs;
122
123 // imported types
124 // FIXME: only import public declarations
125 vector<Decl *> decls = imported_ctx->get_decls();
126 for (auto *t : decls) {
127 if (t->is_type_decl()) {
128 top_ctx()->set_decl(t->get_name(), t);
129 p->_imported_types.push_back(pcast<TypeDecl>(t));
130 }
131 }
132}

191DEFINE_AST_VISITOR_IMPL(TypePrecheck, FunctionDecl) {
192 push_scope(p);
193
194 /// update return type
195 auto *func_type = pcast<FunctionType>(p->get_type());
196 auto *ret_type = check_type(func_type->get_return_type(), p);
197 func_type->set_return_type(ret_type);
198
199 /// type_check_ast args
200 size_t n = p->get_n_args();
201 const auto &arg_decls = p->get_arg_decls();
202 vector<Type *> arg_types(n, nullptr);
203 for (size_t i = 0; i < n; ++i) {
204 visit(arg_decls[i]); /// args will be added to the scope here
205 arg_types[i] = arg_decls[i]->get_type();
206 }
207 func_type->set_arg_types(arg_types); /// update arg types
208
209 pop_scope();
210}

212DEFINE_AST_VISITOR_IMPL(TypePrecheck, StructDecl) {
213 str struct_name = p->get_name();
214
215 auto members = p->get_member_decls();
216 size_t n = members.size();
217 auto *ty = pcast<StructType>(p->get_type());
218 TAN_ASSERT(ty);
219
220 push_scope(p);
221
222 for (size_t i = 0; i < n; ++i) {
223 Expr *m = members[i];
224
225 // Check if m is a struct member declaration before visiting
226 switch (m->get_node_type()) {
227 case ASTNodeType::VAR_DECL:
228 case ASTNodeType::FUNC_DECL:
229 break;
230 case ASTNodeType::ASSIGN: {
231 auto *assign = pcast<Assignment>(m);
232 if (assign->get_lhs()->get_node_type() != ASTNodeType::VAR_DECL)
233 error(ErrorType::SEMANTIC_ERROR, assign, "Expect a member variable declaration");
234 break;
235 }
236 default:
237 error(ErrorType::SEMANTIC_ERROR, p, "Invalid struct member");
238 }
239
240 visit(m);
241
242 /*
243 * DO NOT add unresolved symbol dependency if m's type is a pointer to an unresolved type reference
244 * This allows us to define a struct that holds a pointer to itself, like LinkedList.
245 *
246 * This works because:
247 * 1. m is registered in the unresolved symbol dependency graph so it will be re-analyzed in
248 * the final analysis stage.
249 * 2. Nothing actually directly relies on the type of m. For example, size in bits is always the size of a
250 * pointer.
251 */
252 if (!m->get_type()->is_canonical() && !m->get_type()->is_pointer()) {
253 _package->top_level_symbol_dependency.add_dependency(m, p);
254 }
255
256 if (m->get_node_type() == ASTNodeType::VAR_DECL) { // member variable without initial value
257 str name = pcast<VarDecl>(m)->get_name();
258 p->set_member_index(name, (int)i);
259 (*ty)[i] = m->get_type();
260
261 } else if (m->get_node_type() == ASTNodeType::ASSIGN) { // member variable with an initial value
262 auto *assign = pcast<Assignment>(m);
263 auto decl = pcast<VarDecl>(assign->get_lhs());
264
265 (*ty)[i] = decl->get_type();
266
267 // member name -> index
268 p->set_member_index(decl->get_name(), (int)i);
269
270 // set rhs of assignment as the default value of this member
271 auto *rhs = assign->get_rhs();
272 if (!rhs->is_comptime_known()) {
273 error(ErrorType::SEMANTIC_ERROR, rhs, "Expect the value to be compile time known");
274 }
275 p->set_member_default_val((int)i, rhs);
276
277 } else if (m->get_node_type() == ASTNodeType::FUNC_DECL) { // member functions
278 auto f = pcast<FunctionDecl>(m);
279
280 (*ty)[i] = f->get_type();
281 p->set_member_index(f->get_name(), (int)i);
282 } else {
283 TAN_ASSERT(false);
284 }
285 }
286
287 pop_scope();
288}
