20void TypePrecheck::run_impl(CompilationUnit *cu) {
21 _cu = cu;
22
23 auto *p = cu->ast();
24
25 push_scope(p);
26
27 for (const auto &c : p->get_children()) {
28 if (c->get_node_type() == ASTNodeType::IMPORT) {
29 CALL_AST_VISITOR(Import, c);
30 } else if (c->get_node_type() == ASTNodeType::STRUCT_DECL) {
31 CALL_AST_VISITOR(StructDecl, c);
32 } else if (c->get_node_type() == ASTNodeType::FUNC_DECL) {
33 CALL_AST_VISITOR(FunctionDecl, c);
34 }
35 }
36
37 pop_scope();
38}

40Type *TypePrecheck::check_type_ref(Type *p, SrcLoc loc, ASTBase *node) {
41 TAN_ASSERT(p->is_ref());
42 Type *ret = p;
43
44 const str &referred_name = p->get_typename();
45 auto *decl = search_decl_in_scopes(referred_name);
46 if (decl && decl->is_type_decl()) {
47 ret = decl->get_type();
48 TAN_ASSERT(ret);
49 if (!ret->is_canonical()) {
50 _cu->top_level_symbol_dependency.add_dependency(decl, node);
51 }
52 } else {
53 Error err(_sm->get_filename(), _sm->get_token(loc), fmt::format("Unknown type {}", referred_name));
54 err.raise();
55 }
56
57 return ret;
58}

60Type *TypePrecheck::check_type(Type *p, SrcLoc loc, ASTBase *node) {
61 TAN_ASSERT(p);
62 TAN_ASSERT(node);
63
64 Type *ret = p;
65 if (p->is_ref()) {
66 ret = check_type_ref(p, loc, node);
67 } else if (p->is_pointer()) {
68 auto *pointee = ((PointerType *)p)->get_pointee();
69
70 TAN_ASSERT(pointee);
71 if (pointee->is_ref()) {
72 pointee = check_type_ref(pointee, loc, node);
73 if (pointee->is_canonical()) {
74 ret = Type::GetPointerType(pointee);
75 }
76 }
77 }
78
79 TAN_ASSERT(ret);
80 return ret;
81}

168DEFINE_AST_VISITOR_IMPL(TypePrecheck, StructDecl) {
169 str struct_name = p->get_name();
170
171 auto members = p->get_member_decls();
172 size_t n = members.size();
173 auto *ty = (StructType *)p->get_type();
174 TAN_ASSERT(ty);
175
176 push_scope(p);
177
178 for (size_t i = 0; i < n; ++i) {
179 Expr *m = members[i];
180 visit(m);
181
182 /*
183 * DO NOT add unresolved symbol dependency if m's type is a pointer to an unresolved type reference
184 * This allows us to define a struct that holds a pointer to itself, like LinkedList.
185 *
186 * This works because:
187 * 1. m is registered in the unresolved symbol dependency graph so it will be re-analyzed in
188 * the final analysis stage.
189 * 2. Nothing actually directly relies on the type of m. For example, size in bits is always the size of a
190 * pointer.
191 */
192 if (!m->get_type()->is_canonical() && !m->get_type()->is_pointer()) {
193 _cu->top_level_symbol_dependency.add_dependency(m, p);
194 }
195
196 if (m->get_node_type() == ASTNodeType::VAR_DECL) { /// member variable without initial value
197 str name = ast_cast<VarDecl>(m)->get_name();
198 p->set_member_index(name, i);
199 (*ty)[i] = m->get_type();
200 } else if (m->get_node_type() == ASTNodeType::ASSIGN) { /// member variable with an initial value
201 auto bm = ast_cast<Assignment>(m);
202
203 if (bm->get_lhs()->get_node_type() != ASTNodeType::VAR_DECL) {
204 error(bm, "Expect a member variable declaration");
205 }
206 auto decl = ast_cast<VarDecl>(bm->get_lhs());
207
208 (*ty)[i] = decl->get_type();
209 p->set_member_index(decl->get_name(), i);
210 } else if (m->get_node_type() == ASTNodeType::FUNC_DECL) { /// member functions
211 auto f = ast_cast<FunctionDecl>(m);
212
213 (*ty)[i] = f->get_type();
214 p->set_member_index(f->get_name(), i);
215 } else {
216 error(p, "Invalid struct member");
217 }
218 }
219
220 pop_scope();
221}
