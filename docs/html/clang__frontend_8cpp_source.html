32 static void insertTargetAndModeArgs(const ParsedClangName &NameParts, SmallVectorImpl<const char *> &ArgVector,
33  std::set<str> &SavedStrings) {
34  // Put target and mode arguments at the start of argument list so that
35  // arguments specified in command line could override them. Avoid putting
36  // them at index 0, as an option like '-cc1' must remain the first.
37  int InsertionPoint = 0;
38  if (ArgVector.size() > 0) {
39  ++InsertionPoint;
40  }
41 
42  if (NameParts.DriverMode) {
43  // Add the mode flag to the arguments.
44  ArgVector.insert(ArgVector.begin() + InsertionPoint, GetStableCStr(SavedStrings, NameParts.DriverMode));
45  }
46 
47  if (NameParts.TargetIsValid) {
48  const char *arr[] = {"-target", GetStableCStr(SavedStrings, NameParts.TargetPrefix)};
49  ArgVector.insert(ArgVector.begin() + InsertionPoint, std::begin(arr), std::end(arr));
50  }
51 }

95 int main0(int argc_, const char **argv_) {
96  noteBottomOfStack();
97  llvm::InitLLVM X(argc_, argv_);
98  SmallVector<const char *, 256> argv(argv_, argv_ + argc_);
99  if (llvm::sys::Process::FixupStandardFileDescriptors()) {
100  return 1;
101  }
102 
103  llvm::InitializeAllTargets();
104  auto TargetAndMode = ToolChain::getTargetAndModeFromProgramName(argv[0]);
105 
106  llvm::BumpPtrAllocator A;
107  llvm::StringSaver Saver(A);
108 
109  llvm::cl::TokenizerCallback Tokenizer = &llvm::cl::TokenizeGNUCommandLine;
110  llvm::cl::ExpandResponseFiles(Saver, Tokenizer, argv, false);
111 
112  std::set<str> SavedStrings;
113  str Path = GetExecutablePath(argv[0]);
114  IntrusiveRefCntPtr<DiagnosticOptions> DiagOpts = CreateAndPopulateDiagOpts(argv);
115  TextDiagnosticPrinter *DiagClient = new TextDiagnosticPrinter(llvm::errs(), &*DiagOpts);
116  FixupDiagPrefixExeName(DiagClient, Path);
117  IntrusiveRefCntPtr<DiagnosticIDs> DiagID(new DiagnosticIDs());
118  DiagnosticsEngine Diags(DiagID, &*DiagOpts, DiagClient);
119 
120  if (!DiagOpts->DiagnosticSerializationFile.empty()) {
121  auto SerializedConsumer = clang::serialized_diags::create(DiagOpts->DiagnosticSerializationFile, &*DiagOpts, true);
122  Diags.setClient(new ChainedDiagnosticConsumer(Diags.takeClient(), std::move(SerializedConsumer)));
123  }
124 
125  ProcessWarningOptions(Diags, *DiagOpts, false);
126 
127  Driver TheDriver(Path, llvm::sys::getDefaultTargetTriple(), Diags);
128  TheDriver.setTargetAndMode(TargetAndMode);
129  insertTargetAndModeArgs(TargetAndMode, argv, SavedStrings);
130  SetBackdoorDriverOutputsFromEnvVars(TheDriver);
131 
132  std::unique_ptr<Compilation> C(TheDriver.BuildCompilation(argv));
133  int Res = 1;
134  if (C && !C->containsError()) {
135  SmallVector<pair<int, const Command *>, 4> FailingCommands;
136  Res = TheDriver.ExecuteCompilation(*C, FailingCommands);
137 
138  for (const auto &P : FailingCommands) {
139  int CommandRes = P.first;
140  const Command *FailingCommand = P.second;
141  if (!Res) {
142  Res = CommandRes;
143  }
144 
145  // If result status is < 0, then the driver command signalled an error.
146  // If result status is 70, then the driver command reported a fatal error.
147  // On Windows, abort will return an exit code of 3. In these cases,
148  // generate additional diagnostic information if possible.
149  bool DiagnoseCrash = CommandRes < 0 || CommandRes == 70;
150 #ifdef _WIN32
151  DiagnoseCrash |= CommandRes == 3;
152 #endif
153  if (DiagnoseCrash) {
154  TheDriver.generateCompilationDiagnostics(*C, *FailingCommand);
155  break;
156  }
157  }
158  }
159  Diags.getClient()->finish();
160 
161 #ifdef _WIN32
162  // Exit status should not be negative on Win32, unless abnormal termination.
163  // Once abnormal termination was caught, negative status should not be
164  // propagated.
165  if (Res < 0)
166  Res = 1;
167 #endif
168 
169  /// if we have multiple failing commands, we return the result of the first failing command
170  return Res;
171 }

181 int clang_compile(vector<str> input_files, TanCompilation *config) {
182  vector<const char *> args;
183  size_t n_import = config->import_dirs.size();
184  args.reserve(input_files.size() + 2 * +1);
185  args.push_back("clang");
186  std::for_each(input_files.begin(), input_files.end(), [&args](const auto &s) { args.push_back(s.c_str()); });
187  for (size_t i = 0; i < n_import; ++i) {
188  args.push_back("-I");
189  args.push_back(config->import_dirs[i].c_str());
190  }
191  str opt_level = opt_level_to_string(config->opt_level);
192  args.push_back(opt_level.c_str());
193  args.push_back("-c");
194 
195  llvm::IntrusiveRefCntPtr<clang::DiagnosticIDs> diag_id(new clang::DiagnosticIDs());
196  auto diag_options = new clang::DiagnosticOptions();
197  clang::DiagnosticsEngine diag_engine(diag_id, diag_options,
198  new clang::TextDiagnosticPrinter(llvm::errs(), diag_options));
199  clang::driver::Driver driver(GetExecutablePath("clang"), llvm::sys::getDefaultTargetTriple(), diag_engine);
200 
201  auto *compilation = driver.BuildCompilation(args);
202  SmallVector<pair<int, const Command *>, 0> failing_commands;
203  if (compilation) {
204  return driver.ExecuteCompilation(*compilation, failing_commands);
205  }
206  return 1;
207 }
