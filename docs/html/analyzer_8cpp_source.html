223  void analyze_bop(ASTBase *_p) {
224  auto p = ast_must_cast<BinaryOperator>(_p);
225  Expr *lhs = p->get_lhs();
226  Expr *rhs = p->get_rhs();
227 
228  /// NOTE: do not analyze lhs and rhs just yet, because analyze_assignment
229  /// and analyze_member_access have their own ways of analyzing
230 
231  switch (p->get_op()) {
232  case BinaryOpKind::SUM:
233  case BinaryOpKind::SUBTRACT:
234  case BinaryOpKind::MULTIPLY:
235  case BinaryOpKind::DIVIDE:
236  case BinaryOpKind::MOD: {
237  analyze(lhs);
238  analyze(rhs);
239 
240  int i = TypeSystem::CanImplicitCast(_ctx, lhs->get_type(), rhs->get_type());
241  if (i == -1) {
242  report_error(p, "Cannot perform implicit type conversion");
243  }
244 
245  size_t dominant_idx = static_cast<size_t>(i);
246  p->set_dominant_idx(dominant_idx);
247  ASTType *ty = copy_ty(dominant_idx == 0 ? lhs->get_type() : rhs->get_type());
248  ty->set_is_lvalue(false);
249  p->set_type(ty);
250  break;
251  }
252  case BinaryOpKind::BAND:
253  case BinaryOpKind::LAND:
254  case BinaryOpKind::BOR:
255  case BinaryOpKind::LOR:
256  case BinaryOpKind::XOR:
257  // TODO: implement the type checking of the above operators
258  TAN_ASSERT(false);
259  break;
260  case BinaryOpKind::GT:
261  case BinaryOpKind::GE:
262  case BinaryOpKind::LT:
263  case BinaryOpKind::LE:
264  case BinaryOpKind::EQ:
265  case BinaryOpKind::NE:
266  analyze(lhs);
267  analyze(rhs);
268 
269  p->set_type(ASTType::CreateAndResolve(_ctx, p->get_loc(), Ty::BOOL));
270  break;
271  case BinaryOpKind::MEMBER_ACCESS:
272  analyze_member_access(ast_must_cast<MemberAccess>(p));
273  break;
274  default:
275  TAN_ASSERT(false);
276  break;
277  }
278  }

280  void analyze_uop(ASTBase *_p) {
281  auto p = ast_must_cast<UnaryOperator>(_p);
282  auto rhs = p->get_rhs();
283  analyze(rhs);
284 
285  switch (p->get_op()) {
286  case UnaryOpKind::LNOT:
287  p->set_type(ASTType::CreateAndResolve(_ctx, p->get_loc(), Ty::BOOL));
288  break;
289  case UnaryOpKind::BNOT:
290  p->set_type(copy_ty(rhs->get_type()));
291  break;
292  case UnaryOpKind::ADDRESS_OF: {
293  auto ty = copy_ty(rhs->get_type());
294  p->set_type(ty->get_ptr_to());
295  break;
296  }
297  case UnaryOpKind::PTR_DEREF: {
298  auto *ty = rhs->get_type();
299  if (!ty->is_ptr()) { report_error(ty, "Expect a pointer type"); }
300  ty = copy_ty(ty->get_contained_ty());
301  ty->set_is_lvalue(true);
302  p->set_type(ty);
303  break;
304  }
305  case UnaryOpKind::PLUS:
306  case UnaryOpKind::MINUS: {
307  /// unary plus/minus
308  auto ty = copy_ty(rhs->get_type());
309  ty->set_is_lvalue(false);
310  p->set_type(ty);
311  break;
312  }
313  default:
314  TAN_ASSERT(false);
315  break;
316  }

345  void analyze_assignment(Assignment *p) {
346  Expr *rhs = p->get_rhs();
347  analyze(rhs);
348 
349  auto lhs = p->get_lhs();
350  ASTType *lhs_type = nullptr;
351  switch (lhs->get_node_type()) {
352  case ASTNodeType::ID:
353  analyze(lhs);
354  lhs_type = ast_must_cast<Identifier>(lhs)->get_type();
355  break;
356  case ASTNodeType::STRUCT_DECL:
357  case ASTNodeType::VAR_DECL:
358  case ASTNodeType::ARG_DECL:
359  case ASTNodeType::ENUM_DECL:
360  case ASTNodeType::BOP_OR_UOP:
361  case ASTNodeType::UOP:
362  case ASTNodeType::BOP:
363  analyze(lhs);
364  lhs_type = ast_must_cast<Expr>(lhs)->get_type();
365  break;
366  default:
367  report_error(lhs, "Invalid left-hand operand");
368  }
369 
370  /// if the type of lhs is not set, we deduce it
371  /// NOTE: we only allow type deduction for declarations
372  if (!lhs_type) {
373  lhs_type = copy_ty(rhs->get_type());
374 
375  /// set type of lhs
376  switch (lhs->get_node_type()) {
377  case ASTNodeType::STRUCT_DECL:
378  case ASTNodeType::VAR_DECL:
379  case ASTNodeType::ARG_DECL:
380  case ASTNodeType::ENUM_DECL:
381  ast_must_cast<Decl>(lhs)->set_type(lhs_type);
382  break;
383  default:
384  TAN_ASSERT(false);
385  break;
386  }
387  /// analyze again just to make sure
388  analyze(lhs);
389  }
390 
391  if (TypeSystem::CanImplicitCast(_ctx, lhs_type, rhs->get_type()) != 0) {
392  report_error(p, "Cannot perform implicit type conversion");
393  }
394  p->set_type(lhs_type);
395  }

461  void analyze_intrinsic(ASTBase *_p) {
462  auto p = ast_must_cast<Intrinsic>(_p);
463  auto c = p->get_sub();
464 
465  /// name
466  str name;
467  switch (c->get_node_type()) {
468  case ASTNodeType::FUNC_CALL:
469  name = ast_must_cast<FunctionCall>(c)->get_name();
470  break;
471  case ASTNodeType::ID:
472  name = ast_must_cast<Identifier>(c)->get_name();
473  break;
474  default:
475  TAN_ASSERT(false);
476  break;
477  }
478 
479  /// search for the intrinsic type
480  auto q = Intrinsic::intrinsics.find(name);
481  if (q == Intrinsic::intrinsics.end()) {
482  report_error(p, "Invalid intrinsic");
483  }
484  p->set_intrinsic_type(q->second);
485 
486  auto void_type = ASTType::CreateAndResolve(_ctx, p->get_loc(), Ty::VOID);
487  switch (p->get_intrinsic_type()) {
488  case IntrinsicType::STACK_TRACE:
489  case IntrinsicType::ABORT:
490  case IntrinsicType::NOOP: {
491  analyze(p->get_sub());
492  p->set_type(void_type);
493  break;
494  }
495  case IntrinsicType::LINENO: {
496  auto sub = IntegerLiteral::Create(p->get_loc(), _sm->get_line(p->get_loc()), true);
497  auto type = ASTType::CreateAndResolve(_ctx, p->get_loc(), TY_OR3(Ty::INT, Ty::UNSIGNED, Ty::BIT32));
498  sub->set_type(type);
499  p->set_type(type);
500  p->set_sub(sub);
501  break;
502  }
503  case IntrinsicType::FILENAME: {
504  auto sub = StringLiteral::Create(p->get_loc(), _ctx->_filename);
505  auto type = ASTType::CreateAndResolve(_ctx, p->get_loc(), Ty::STRING);
506  sub->set_type(type);
507  p->set_type(type);
508  p->set_sub(sub);
509  break;
510  }
511  case IntrinsicType::GET_DECL: {
512  TAN_ASSERT(c->get_node_type() == ASTNodeType::FUNC_CALL);
513  auto *func_call = ast_must_cast<FunctionCall>(c);
514  if (func_call->get_n_args() != 1) {
515  report_error(func_call, "Expect the number of args to be 1");
516  }
517  auto *target = func_call->get_arg(0);
518  auto *source_str =
519  ASTBuilder::CreateStringLiteral(_ctx, p->get_loc(), _ctx->get_source_manager()->get_source_code(target));
520  p->set_sub(source_str);
521  p->set_type(source_str->get_type());
522  break;
523  }
524  case IntrinsicType::COMP_PRINT: {
525  p->set_type(void_type);
526 
527  auto func_call = ast_must_cast<FunctionCall>(c);
528  auto args = func_call->_args;
529 
530  if (args.size() != 1 || args[0]->get_node_type() != ASTNodeType::STRING_LITERAL) {
531  report_error(p, "Invalid call to compprint, one argument with type 'str' required");
532  }
533  str msg = ast_must_cast<StringLiteral>(args[0])->get_value();
534  std::cout << fmt::format("Message ({}): {}\n", _ctx->get_source_location_str(p), msg);
535  break;
536  }
537  default:
538  report_error(p, "Unknown intrinsic");
539  }
540  }

673  Expr *lhs = p->get_lhs();
674  analyze(lhs);
675  Expr *rhs = p->get_rhs();

696  void analyze_struct_decl(ASTBase *_p) {
697  auto p = ast_must_cast<StructDecl>(_p);
698 
699  ASTType *ty = nullptr;
700  str struct_name = p->get_name();
701  _ctx->add_type_decl(struct_name, p);
702 
703  /// check if struct name is in conflicts of variable/function names
704  /// or if there's a forward declaration
705  ASTType *prev = _ctx->get_type_decl(struct_name)->get_type();
706  if (prev) {
707  if (prev->get_node_type() != ASTNodeType::STRUCT_DECL) { /// fwd decl
708  ty = prev;
709  } else { /// conflict
710  report_error(p, "Cannot redeclare type as a struct");
711  }
712  } else { /// no fwd decl
713  ty = ASTType::Create(_ctx, p->get_loc());
714  ty->set_ty(Ty::STRUCT);
715  ty->set_constructor(StructConstructor::Create(ty));
716  _ctx->add_type_decl(struct_name, p); /// add self to current scope
717  }
718  ty->set_type_name(struct_name);
719 
720  /// resolve member names and types
721  auto member_decls = p->get_member_decls(); // size is 0 if no struct body
722  size_t n = member_decls.size();
723  ty->get_sub_types().reserve(n);
724  for (size_t i = 0; i < n; ++i) {
725  Expr *m = member_decls[i];
726  analyze(m);
727 
728  if (m->get_node_type() == ASTNodeType::VAR_DECL) { /// member variable without initial value
729  /// fill members
730  str name = ast_must_cast<VarDecl>(m)->get_name();
731  p->set_member_index(name, i);
732  ty->get_sub_types().push_back(m->get_type());
733  }
734  /// member variable with an initial value
735  else if (m->get_node_type() == ASTNodeType::ASSIGN) {
736  auto bm = ast_must_cast<Assignment>(m);
737  auto init_val = bm->get_rhs();
738 
739  if (bm->get_lhs()->get_node_type() != ASTNodeType::VAR_DECL) {
740  report_error(bm, "Expect a member variable declaration");
741  }
742  auto decl = ast_must_cast<VarDecl>(bm->get_lhs());
743 
744  /// fill members
745  ty->get_sub_types().push_back(decl->get_type());
746  p->set_member_index(decl->get_name(), i);
747 
748  /// initial values
749  if (!init_val->is_comptime_known()) {
750  report_error(p, "Initial value of a member variable must be compile-time known");
751  }
752  auto *ctr = cast_ptr<StructConstructor>(ty->get_constructor());
753  ctr->get_member_constructors().push_back(BasicConstructor::Create(ast_must_cast<CompTimeExpr>(init_val)));
754  } else if (m->get_node_type() == ASTNodeType::FUNC_DECL) {
755  auto f = ast_must_cast<FunctionDecl>(m);
756 
757  /// fill members
758  ty->get_sub_types().push_back(f->get_type());
759  p->set_member_index(f->get_name(), i);
760  } else {
761  report_error(p, "Invalid struct member");
762  }
763  }
764  resolve_ty(ty);
765  p->set_type(ty);
766  }
