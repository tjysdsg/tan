112  /**
113  * \brief Find the type that operands of a BOP should promote to, and add a Cast node to the AST.
114  * Raise an error if can't find a valid type promotion.
115  * \note This could modify the lhs or rhs of `bop`, make sure to update the references to any of them after calling.
116  * \return The promoted type.
117  */
118  Type *auto_promote_bop_operand_types(BinaryOperator *bop) {
119  auto *lhs = bop->get_lhs();
120  auto *rhs = bop->get_rhs();
121  auto *lhs_type = lhs->get_type();
122  auto *rhs_type = rhs->get_type();
123 
124  auto *promoted_type = ImplicitTypePromote(lhs_type, rhs_type);
125  if (!promoted_type) {
126  error(bop, fmt::format("Cannot find a valid type promotion between {} and {}", lhs_type->get_typename(),
127  rhs_type->get_typename()));
128  }
129 
130  TAN_ASSERT(promoted_type == lhs_type || promoted_type == rhs_type);
131  if (promoted_type != lhs_type) {
132  auto *cast = Cast::Create(bop->loc());
133  cast->set_lhs(lhs);
134  cast->set_type(promoted_type);
135  bop->set_lhs(cast);
136  } else {
137  auto *cast = Cast::Create(bop->loc());
138  cast->set_lhs(rhs);
139  cast->set_type(promoted_type);
140  bop->set_rhs(cast);
141  }
142 
143  return promoted_type;
144  }

146  FunctionDecl *search_function_callee(FunctionCall *p) {
147  const str &name = p->get_name();
148  const vector<Expr *> &args = p->_args;
149 
150  FunctionDecl *ret = nullptr;
151  auto func_candidates = _ctx->get_functions(name);
152  /// find a valid function overload to call
153  for (const auto &f : func_candidates) {
154  size_t n = f->get_n_args();
155  if (n != args.size()) {
156  continue;
157  }
158 
159  /// check if argument types match (return type not checked)
160  /// allow implicit cast from actual arguments to expected arguments
161  bool good = true;
162  int cost = 0; /// number of implicit type conversion of arguments needed
163  for (size_t i = 0; i < n; ++i) {
164  auto *actual_type = args[i]->get_type();
165  auto *expected_type = f->get_arg_type(i);
166 
167  if (actual_type != expected_type) {
168  ++cost;
169 
170  if (!CanImplicitlyConvert(actual_type, expected_type)) {
171  good = false;
172  break;
173  }
174  }
175  }
176 
177  /// remember valid candidate(s) and check for ambiguity
178  if (good) {
179  /// if there is an exact match, use it
180  if (cost == 0) {
181  ret = f;
182  break;
183  }
184 
185  if (ret) {
186  // TODO: print all valid candidates
187  error(p, "Ambiguous function call: " + name);
188  }
189  ret = f;
190  }
191  }
192 
193  if (!ret) {
194  Error err(_ctx->get_filename(), _ctx->get_source_manager()->get_token(p->loc()),
195  "Unknown function call: " + name);
196  err.raise();
197  }
198  return ret;
199  }

319  void analyze_bop(ASTBase *_p) {
320  auto p = ast_cast<BinaryOperator>(_p);
321  Expr *lhs = p->get_lhs();
322  Expr *rhs = p->get_rhs();
323 
324  /// NOTE: do not analyze lhs and rhs just yet, because analyze_assignment
325  /// and analyze_member_access have their own ways of analyzing
326 
327  switch (p->get_op()) {
328  case BinaryOpKind::SUM:
329  case BinaryOpKind::SUBTRACT:
330  case BinaryOpKind::MULTIPLY:
331  case BinaryOpKind::DIVIDE:
332  case BinaryOpKind::BAND:
333  case BinaryOpKind::BOR:
334  case BinaryOpKind::MOD: {
335  analyze(lhs);
336  analyze(rhs);
337  p->set_type(auto_promote_bop_operand_types(p));
338  break;
339  }
340  case BinaryOpKind::LAND:
341  case BinaryOpKind::LOR:
342  case BinaryOpKind::XOR: {
343  analyze(lhs);
344  analyze(rhs);
345 
346  // check if both operators are bool
347  auto *bool_type = PrimitiveType::GetBoolType();
348  p->set_lhs(create_implicit_conversion(lhs, bool_type));
349  p->set_rhs(create_implicit_conversion(rhs, bool_type));
350  p->set_type(bool_type);
351  break;
352  }
353  case BinaryOpKind::GT:
354  case BinaryOpKind::GE:
355  case BinaryOpKind::LT:
356  case BinaryOpKind::LE:
357  case BinaryOpKind::EQ:
358  case BinaryOpKind::NE: {
359  analyze(lhs);
360  analyze(rhs);
361  auto_promote_bop_operand_types(p);
362  p->set_type(PrimitiveType::GetBoolType());
363  break;
364  }
365  case BinaryOpKind::MEMBER_ACCESS:
366  analyze_member_access(ast_cast<MemberAccess>(p));
367  break;
368  default:
369  TAN_ASSERT(false);
370  }

373  void analyze_uop(ASTBase *_p) {
374  auto *p = ast_cast<UnaryOperator>(_p);
375  auto *rhs = p->get_rhs();
376  analyze(rhs);
377 
378  auto *rhs_type = rhs->get_type();
379  switch (p->get_op()) {
380  case UnaryOpKind::LNOT:
381  rhs = create_implicit_conversion(rhs, Type::GetBoolType());
382  p->set_rhs(rhs);
383  p->set_type(PrimitiveType::GetBoolType());
384  break;
385  case UnaryOpKind::BNOT:
386  if (!rhs_type->is_int()) {
387  error(rhs, "Expect an integer type");
388  }
389  p->set_type(rhs_type);
390  break;
391  case UnaryOpKind::ADDRESS_OF:
392  p->set_type(Type::GetPointerType(rhs_type));
393  break;
394  case UnaryOpKind::PTR_DEREF:
395  if (!rhs_type->is_pointer()) {
396  error(rhs, "Expect a pointer type");
397  }
398  TAN_ASSERT(rhs->is_lvalue());
399  p->set_lvalue(true);
400  p->set_type(((PointerType *)rhs_type)->get_pointee());
401  break;
402  case UnaryOpKind::PLUS:
403  case UnaryOpKind::MINUS: /// unary plus/minus
404  if (!(rhs_type->is_int() || rhs_type->is_float())) {
405  error(rhs, "Expect an numerical type");
406  }
407  p->set_type(rhs_type);
408  break;
409  default:
410  TAN_ASSERT(false);
411  }

421  void analyze_assignment(ASTBase *_p) {
422  auto *p = ast_cast<Assignment>(_p);
423 
424  Expr *rhs = p->get_rhs();
425  analyze(rhs);
426 
427  auto *lhs = p->get_lhs();
428  Type *lhs_type = nullptr;
429  switch (lhs->get_node_type()) {
430  case ASTNodeType::ID:
431  analyze(lhs);
432  lhs_type = ast_cast<Identifier>(lhs)->get_type();
433  break;
434  case ASTNodeType::VAR_DECL:
435  case ASTNodeType::ARG_DECL:
436  case ASTNodeType::BOP_OR_UOP:
437  case ASTNodeType::UOP:
438  case ASTNodeType::BOP:
439  analyze(lhs);
440  lhs_type = ast_cast<Expr>(lhs)->get_type();
441  break;
442  default:
443  error(lhs, "Invalid left-hand operand");
444  }
445 
446  /// if the type of lhs is not set, we deduce it
447  /// NOTE: we only allow type deduction for variable declarations
448  if (!lhs_type) {
449  lhs_type = rhs->get_type();
450 
451  /// set type of lhs
452  switch (lhs->get_node_type()) {
453  case ASTNodeType::VAR_DECL:
454  ast_cast<Decl>(lhs)->set_type(lhs_type);
455  break;
456  default:
457  TAN_ASSERT(false);
458  }
459  /// analyze again just to make sure
460  analyze(lhs);
461  }
462 
463  rhs = create_implicit_conversion(rhs, lhs_type);
464  p->set_rhs(rhs);
465 
466  p->set_lvalue(true);
467  p->set_type(lhs_type);
468  }

594  void analyze_intrinsic(ASTBase *_p) {
595  auto p = ast_cast<Intrinsic>(_p);
596  auto c = p->get_sub();
597 
598  /// name
599  str name;
600  switch (c->get_node_type()) {
601  case ASTNodeType::FUNC_CALL: {
602  auto *func_call = ast_cast<FunctionCall>(c);
603  name = func_call->get_name();
604  find_and_assign_intrinsic_type(p, name);
605  analyze_intrinsic_func_call(p, func_call);
606  return;
607  }
608  case ASTNodeType::ID:
609  name = ast_cast<Identifier>(c)->get_name();
610  break;
611  default:
612  name = p->get_name();
613  break;
614  }
615  TAN_ASSERT(!name.empty());
616  find_and_assign_intrinsic_type(p, name);
617 
618  switch (p->get_intrinsic_type()) {
619  case IntrinsicType::LINENO: {
620  auto sub = IntegerLiteral::Create(p->loc(), _sm->get_line(p->loc()), true);
621  auto type = PrimitiveType::GetIntegerType(32, true);
622  sub->set_type(type);
623  p->set_type(type);
624  p->set_sub(sub);
625  break;
626  }
627  case IntrinsicType::FILENAME: {
628  auto sub = StringLiteral::Create(p->loc(), _ctx->_filename);
629  auto type = Type::GetStringType();
630  sub->set_type(type);
631  p->set_type(type);
632  p->set_sub(sub);
633  break;
634  }
635  case IntrinsicType::TEST_COMP_ERROR: {
636  // FIXME: avoid setjmp and longjmp
637  bool error_catched = false;
638  std::jmp_buf buf;
639  if (setjmp(buf) > 0) {
640  error_catched = true;
641  } else {
642  auto error_catcher = ErrorCatcher((const ErrorCatcher::callback_t &)[&](str) { longjmp(buf, 1); });
643  Error::CatchErrors(&error_catcher);
644  analyze(p->get_sub());
645  }
646 
647  Error::ResetErrorCatcher();
648  if (!error_catched) {
649  error(p, "Expect a compile error");
650  }
651  break;
652  }
653  default:
654  TAN_ASSERT(false);
655  }

731  /// ASSUMES lhs has been already analyzed, while rhs has not
732  void analyze_bracket_access(MemberAccess *p, Expr *lhs, Expr *rhs) {
733  analyze(rhs);
734 
735  if (!lhs->is_lvalue()) {
736  error(p, "Expect lhs to be an lvalue");
737  }
738 
739  auto *lhs_type = lhs->get_type();
740  if (!(lhs_type->is_pointer() || lhs_type->is_array() || lhs_type->is_string())) {
741  error(p, "Expect a type that supports bracket access");
742  }
743  if (!rhs->get_type()->is_int()) {
744  error(rhs, "Expect an integer");
745  }
746 
747  Type *sub_type = nullptr;
748  if (lhs_type->is_pointer()) {
749  sub_type = ((PointerType *)lhs_type)->get_pointee();
750  } else if (lhs_type->is_array()) {
751  auto *array_type = (ArrayType *)lhs_type;
752  sub_type = array_type->get_element_type();
753  /// check if array index is out-of-bound
754  if (rhs->get_node_type() == ASTNodeType::INTEGER_LITERAL) {
755  uint64_t size = ast_cast<IntegerLiteral>(rhs)->get_value();
756  if (lhs->get_type()->is_array() && (int)size >= array_type->get_size()) {
757  error(p, fmt::format("Index {} out of bound, the array size is {}", std::to_string(size),
758  std::to_string(array_type->get_size())));
759  }
760  }
761  } else if (lhs_type->is_string()) {
762  sub_type = Type::GetCharType();
763  }
764 
765  p->set_type(sub_type);
766  }

809  void analyze_member_access(MemberAccess *p) {
810  Expr *lhs = p->get_lhs();
811  analyze(lhs);
812  Expr *rhs = p->get_rhs();
813 
814  if (rhs->get_node_type() == ASTNodeType::FUNC_CALL) { /// method call
815  p->_access_type = MemberAccess::MemberAccessMemberFunction;
816  auto func_call = ast_cast<FunctionCall>(rhs);
817  analyze_member_func_call(p, lhs, func_call);
818  } else if (p->_access_type == MemberAccess::MemberAccessBracket) {
819  analyze_bracket_access(p, lhs, rhs);
820  } else if (rhs->get_node_type() == ASTNodeType::ID) { /// member variable or enum
821  if (lhs->get_type()->is_enum()) {
822  p->_access_type = MemberAccess::MemberAccessEnumValue;
823  analyze_enum_member_access(p, lhs, rhs);
824  } else { /// member variable
825  p->_access_type = MemberAccess::MemberAccessMemberVariable;
826  analyze_member_access_member_variable(p, lhs, rhs);
827  }
828  } else {
829  error(p, "Invalid right-hand operand");
830  }
831  }

833  void analyze_struct_decl(ASTBase *_p) {
834  auto p = ast_cast<StructDecl>(_p);
835 
836  /// check if struct name is in conflicts of variable/function names
837  /// or if there's a forward declaration
838  str struct_name = p->get_name();
839  auto *prev_decl = _ctx->get_type_decl(struct_name);
840  if (prev_decl) {
841  if (!(prev_decl->get_node_type() == ASTNodeType::STRUCT_DECL &&
842  ast_cast<StructDecl>(prev_decl)->is_forward_decl())) { /// conflict
843  error(p, "Cannot redeclare type as a struct");
844  }
845  }
846 
847  /// resolve member names and types
848  auto member_decls = p->get_member_decls(); // size is 0 if no struct body
849  size_t n = member_decls.size();
850  vector<Type *> child_types(n, nullptr);
851  for (size_t i = 0; i < n; ++i) {
852  Expr *m = member_decls[i];
853  analyze(m);
854 
855  if (m->get_node_type() == ASTNodeType::VAR_DECL) { /// member variable without initial value
856  /// fill members
857  str name = ast_cast<VarDecl>(m)->get_name();
858  p->set_member_index(name, i);
859  child_types[i] = m->get_type();
860  } else if (m->get_node_type() == ASTNodeType::ASSIGN) { /// member variable with an initial value
861  auto bm = ast_cast<Assignment>(m);
862  auto init_val = bm->get_rhs();
863 
864  if (bm->get_lhs()->get_node_type() != ASTNodeType::VAR_DECL) {
865  error(bm, "Expect a member variable declaration");
866  }
867  auto decl = ast_cast<VarDecl>(bm->get_lhs());
868 
869  /// fill members
870  child_types[i] = decl->get_type();
871  p->set_member_index(decl->get_name(), i);
872 
873  /// initial values
874  if (!init_val->is_comptime_known()) {
875  error(p, "Initial value of a member variable must be compile-time known");
876  }
877  // TODO IMPORTANT: auto *ctr = cast_ptr<StructConstructor>(ty->get_constructor());
878  // ctr->get_member_constructors().push_back(BasicConstructor::Create(ast_cast<CompTimeExpr>(init_val)));
879  } else if (m->get_node_type() == ASTNodeType::FUNC_DECL) { /// member functions
880  auto f = ast_cast<FunctionDecl>(m);
881 
882  /// fill members
883  child_types[i] = f->get_type();
884  p->set_member_index(f->get_name(), i);
885  } else {
886  error(p, "Invalid struct member");
887  }
888  }
889 
890  /// register type to context
891  auto *ty = Type::GetStructType(struct_name, child_types);
892  _ctx->add_type_decl(struct_name, p);
893  p->set_type(ty);
894  }
