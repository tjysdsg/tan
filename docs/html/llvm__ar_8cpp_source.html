76 static bool comparePaths(StringRef Path1, StringRef Path2) {
77 /**
78  * when on Windows this function calls CompareStringOrdinal
79  * as Windows file paths are case-insensitive.
80  * CompareStringOrdinal compares two Unicode strings for
81  * binary equivalence and allows for case insensitivity.
82  */
83 #ifdef _WIN32
84  SmallVector<wchar_t, 128> WPath1, WPath2;
85  failIfError(sys::path::widenPath(normalizePath(Path1), WPath1));
86  failIfError(sys::path::widenPath(normalizePath(Path2), WPath2));
87 
88  return CompareStringOrdinal(WPath1.data(), WPath1.size(), WPath2.data(), WPath2.size(), true) == CSTR_EQUAL;
89 #else
90  return normalizePath(Path1) == normalizePath(Path2);
91 #endif
92 }

109 static InsertAction computeInsertAction(StringRef Name, std::vector<str>::iterator &Pos) {
110  auto MI = find_if(Members, [Name](StringRef Path) { return comparePaths(Name, Path); });
111  if (MI == Members.end()) {
112  return IA_AddOldMember;
113  }
114  Pos = MI;
115  if (RelPos.empty()) {
116  return IA_AddNewMember;
117  }
118  return IA_MoveNewMember;
119 }

121 static std::vector<NewArchiveMember> computeNewArchiveMembers(object::Archive *OldArchive) {
122  /// we have to walk this twice and computing it is not trivial, so creating an
123  /// explicit std::vector is actually fairly efficient
124  std::vector<NewArchiveMember> Ret;
125  std::vector<NewArchiveMember> Moved;
126  int InsertPos = -1;
127  if (OldArchive) {
128  Error Err = Error::success();
129  for (auto &Child : OldArchive->children(Err)) {
130  int Pos = (int)Ret.size();
131  Expected<StringRef> NameOrErr = Child.getName();
132  failIfError(NameOrErr.takeError());
133  std::string Name = std::string(NameOrErr.get());
134  if (comparePaths(Name, RelPos)) {
135  InsertPos = Pos + 1;
136  }
137  auto MemberI = Members.end();
138  InsertAction Action = computeInsertAction(Name, MemberI);
139  switch (Action) {
140  case IA_AddOldMember:
141  addChildMember(Ret, Child);
142  break;
143  case IA_AddNewMember:
144  addMember(Ret, *MemberI);
145  break;
146  case IA_MoveOldMember:
147  addChildMember(Moved, Child);
148  break;
149  case IA_MoveNewMember:
150  addMember(Moved, *MemberI);
151  break;
152  default:
153  break;
154  }
155  if (MemberI != Members.end()) {
156  Members.erase(MemberI);
157  }
158  }
159  failIfError(std::move(Err));
160  }
161 
162  if (!RelPos.empty() && InsertPos == -1) {
163  fail("insertion point not found");
164  }
165  if (RelPos.empty()) {
166  InsertPos = (int)Ret.size();
167  }
168 
169  assert(unsigned(InsertPos) <= Ret.size());
170  int Pos = InsertPos;
171  for (auto &M : Moved) {
172  Ret.insert(Ret.begin() + Pos, std::move(M));
173  ++Pos;
174  }
175 
176  std::vector<NewArchiveMember> NewMembers;
177  for (auto &Member : Members) {
178  addMember(NewMembers, Member);
179  }
180  Ret.reserve(Ret.size() + NewMembers.size());
181  std::move(NewMembers.begin(), NewMembers.end(), std::inserter(Ret, std::next(Ret.begin(), InsertPos)));
182  return Ret;
183 }

189 static object::Archive::Kind getKindFromMember(const NewArchiveMember &Member) {
190  auto MemBufferRef = Member.Buf->getMemBufferRef();
191  Expected<std::unique_ptr<object::ObjectFile>> OptionalObject = object::ObjectFile::createObjectFile(MemBufferRef);
192 
193  if (OptionalObject) {
194  return isa<object::MachOObjectFile>(**OptionalObject) ? object::Archive::K_DARWIN : object::Archive::K_GNU;
195  }
196 
197  // squelch the error in case we had a non-object file
198  consumeError(OptionalObject.takeError());
199 
200  // If we're adding a bitcode file to the archive, detect the Archive kind
201  // based on the target triple.
202  LLVMContext Context;
203  if (identify_magic(MemBufferRef.getBuffer()) == file_magic::bitcode) {
204  if (auto ObjOrErr = object::SymbolicFile::createSymbolicFile(MemBufferRef, file_magic::bitcode, &Context)) {
205  auto &IRObject = cast<object::IRObjectFile>(**ObjOrErr);
206  return Triple(IRObject.getTargetTriple()).isOSDarwin() ? object::Archive::K_DARWIN : object::Archive::K_GNU;
207  } else {
208  // Squelch the error in case this was not a SymbolicFile.
209  consumeError(ObjOrErr.takeError());
210  }
211  }
212 
213  return getDefaultForHost();
214 }

216 static void performWriteOperation(object::Archive *OldArchive, std::unique_ptr<MemoryBuffer> OldArchiveBuf) {
217  std::vector<NewArchiveMember> NewMembers = computeNewArchiveMembers(OldArchive);
218  object::Archive::Kind Kind;
219  if (OldArchive) {
220  Kind = OldArchive->kind();
221  } else {
222  Kind = !NewMembers.empty() ? getKindFromMember(NewMembers.front()) : getDefaultForHost();
223  }
224  Error E = writeArchive(ArchiveName, NewMembers, Symtab, Kind, Deterministic, false, std::move(OldArchiveBuf));
225  failIfError(std::move(E), ArchiveName);
226 }

228 extern int llvm_ar_create_static_lib(const str &archive_name, const vector<str> &objects) {
229  ArchiveName = archive_name;
230  Members = objects;
231  llvm::InitializeAllTargetInfos();
232  llvm::InitializeAllTargetMCs();
233  llvm::InitializeAllAsmParsers();
234  /// create or open the archive object.
235  ErrorOr<std::unique_ptr<MemoryBuffer>> Buf = MemoryBuffer::getFile(archive_name, -1, false);
236  std::error_code EC = Buf.getError();
237  if (EC && EC != errc::no_such_file_or_directory) {
238  fail("error opening '" + archive_name + "': " + EC.message());
239  }
240  if (!EC) {
241  Error Err = Error::success();
242  object::Archive Archive(Buf.get()->getMemBufferRef(), Err);
243  failIfError(std::move(Err), "unable to load '" + archive_name + "'");
244  performWriteOperation(&Archive, std::move(Buf.get()));
245  return 0;
246  }
247  assert(EC == errc::no_such_file_or_directory);
248  performWriteOperation(nullptr, nullptr);
249  return 0;
250 }
